40 struct cascadeproducer {
41  Produces<aod::CascData> cascdata;

43  OutputObj<TH1F> hEventCounter{TH1F("hEventCounter", "", 1, 0, 1)};
44  OutputObj<TH1F> hCascCandidate{TH1F("hCascCandidate", "", 10, 0, 10)};

46  //Configurables
47  Configurable<double> d_bz{"d_bz", -5.0, "bz field"};
48  Configurable<double> d_UseAbsDCA{"d_UseAbsDCA", kTRUE, "Use Abs DCAs"};

52  template <typename T, typename U>
53  auto getdcaXY(const T& track, const U& coll)
54  {
55  //Calculate DCAs
56  auto sinAlpha = sin(track.alpha());
57  auto cosAlpha = cos(track.alpha());
58  auto globalX = track.x() * cosAlpha - track.y() * sinAlpha;
59  auto globalY = track.x() * sinAlpha + track.y() * cosAlpha;
60  return sqrt(pow((globalX - coll[0]), 2) +
61  pow((globalY - coll[1]), 2));
62  }
63 
64  void process(aod::Collision const& collision, aod::V0s const& V0s, aod::Cascades const& Cascades, aod::FullTracks const& trackss)
65  {
66  //Define o2 fitter, 2-prong
67  o2::vertexing::DCAFitterN<2> fitterV0, fitterCasc;
68  fitterV0.setBz(d_bz);
69  fitterV0.setPropagateToPCA(true);
70  fitterV0.setMaxR(200.);
71  fitterV0.setMinParamChange(1e-3);
72  fitterV0.setMinRelChi2Change(0.9);
73  fitterV0.setMaxDZIni(1e9);
74  fitterV0.setMaxChi2(1e9);
75  fitterV0.setUseAbsDCA(d_UseAbsDCA);
76 
77  fitterCasc.setBz(d_bz);
78  fitterCasc.setPropagateToPCA(true);
79  fitterCasc.setMaxR(200.);
80  fitterCasc.setMinParamChange(1e-3);
81  fitterCasc.setMinRelChi2Change(0.9);
82  fitterCasc.setMaxDZIni(1e9);
83  fitterCasc.setMaxChi2(1e9);
84  fitterCasc.setUseAbsDCA(d_UseAbsDCA);
85 
86  hEventCounter->Fill(0.5);
87  std::array<float, 3> pVtx = {collision.posX(), collision.posY(), collision.posZ()};
88 
89  for (auto& casc : Cascades) {
90  auto charge = -1;
91  std::array<float, 3> pos = {0.};
92  std::array<float, 3> posXi = {0.};
93  std::array<float, 3> pvecpos = {0.};
94  std::array<float, 3> pvecneg = {0.};
95  std::array<float, 3> pvecbach = {0.};
96 
97  hCascCandidate->Fill(0.5);
98 
99  //Acquire basic tracks
100  auto pTrack = getTrackParCov(casc.v0().posTrack());
101  auto nTrack = getTrackParCov(casc.v0().negTrack());
102  auto bTrack = getTrackParCov(casc.bachelor());
103  if (casc.bachelor().signed1Pt() > 0)
104  charge = +1;
105 
106  int nCand = fitterV0.process(pTrack, nTrack);
107  if (nCand != 0) {
108  fitterV0.propagateTracksToVertex();
109  hCascCandidate->Fill(1.5);
110  const auto& v0vtx = fitterV0.getPCACandidate();
111  for (int i = 0; i < 3; i++)
112  pos[i] = v0vtx[i];
113 
114  std::array<float, 21> cov0 = {0};
115  std::array<float, 21> cov1 = {0};
116  std::array<float, 21> covV0 = {0};
117 
118  //Covariance matrix calculation
119  const int momInd[6] = {9, 13, 14, 18, 19, 20}; // cov matrix elements for momentum component
120  fitterV0.getTrack(0).getPxPyPzGlo(pvecpos);
121  fitterV0.getTrack(1).getPxPyPzGlo(pvecneg);
122  fitterV0.getTrack(0).getCovXYZPxPyPzGlo(cov0);
123  fitterV0.getTrack(1).getCovXYZPxPyPzGlo(cov1);
124  for (int i = 0; i < 6; i++) {
125  int j = momInd[i];
126  covV0[j] = cov0[j] + cov1[j];
127  }
128  auto covVtxV0 = fitterV0.calcPCACovMatrix();
129  covV0[0] = covVtxV0(0, 0);
130  covV0[1] = covVtxV0(1, 0);
131  covV0[2] = covVtxV0(1, 1);
132  covV0[3] = covVtxV0(2, 0);
133  covV0[4] = covVtxV0(2, 1);
134  covV0[5] = covVtxV0(2, 2);
135 
136  const std::array<float, 3> vertex = {(float)v0vtx[0], (float)v0vtx[1], (float)v0vtx[2]};
137  const std::array<float, 3> momentum = {pvecpos[0] + pvecneg[0], pvecpos[1] + pvecneg[1], pvecpos[2] + pvecneg[2]};
138 
139  auto tV0 = o2::track::TrackParCov(vertex, momentum, covV0, 0);
140  tV0.setQ2Pt(0); //No bending, please
141  int nCand2 = fitterCasc.process(tV0, bTrack);
142  if (nCand2 != 0) {
143  fitterCasc.propagateTracksToVertex();
144  hCascCandidate->Fill(2.5);
145  const auto& cascvtx = fitterCasc.getPCACandidate();
146  for (int i = 0; i < 3; i++)
147  posXi[i] = cascvtx[i];
148  fitterCasc.getTrack(1).getPxPyPzGlo(pvecbach);
149  } //end if cascade recoed
150  } //end if v0 recoed
151  //Fill table, please
152  cascdata(charge, posXi[0], posXi[1], posXi[2], pos[0], pos[1], pos[2],
153  pvecpos[0], pvecpos[1], pvecpos[2],
154  pvecneg[0], pvecneg[1], pvecneg[2],
155  pvecbach[0], pvecbach[1], pvecbach[2],
156  fitterV0.getChi2AtPCACandidate(), fitterCasc.getChi2AtPCACandidate(),
157  getdcaXY(casc.v0().posTrack(), pVtx),
158  getdcaXY(casc.v0().negTrack(), pVtx),
159  getdcaXY(casc.bachelor(), pVtx));
160  }
161  }
162 };
