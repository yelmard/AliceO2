72 template <typename T, typename U, typename V>
73 auto getRotatedCovMatrixXX(const T& matrix, U phi, V theta)
74 {
75  auto cp = std::cos(phi);
76  auto sp = std::sin(phi);
77  auto ct = std::cos(theta);
78  auto st = std::sin(theta);
79  return matrix[0] * cp * cp * ct * ct // covXX
80  + matrix[1] * 2. * cp * sp * ct * ct // covXY
81  + matrix[2] * sp * sp * ct * ct // covYY
82  + matrix[3] * 2. * cp * ct * st // covXZ
83  + matrix[4] * 2. * sp * ct * st // covYZ
84  + matrix[5] * st * st; // covZZ
85 }
