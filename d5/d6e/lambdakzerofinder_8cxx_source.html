72  template <typename T, typename U>
73  auto getdcaXY(const T& track, const U& coll)
74  {
75  //Calculate DCAs
76  auto sinAlpha = sin(track.alpha());
77  auto cosAlpha = cos(track.alpha());
78  auto globalX = track.x() * cosAlpha - track.y() * sinAlpha;
79  auto globalY = track.x() * sinAlpha + track.y() * cosAlpha;
80  return sqrt(pow((globalX - coll[0]), 2) +
81  pow((globalY - coll[1]), 2));
82  }

84  void process(aod::Collision const& collision,
85  soa::Join<aod::FullTracks, aod::TracksExtended> const& tracks)
86  {
87  //Define o2 fitter, 2-prong
88  o2::vertexing::DCAFitterN<2> fitter;
89  fitter.setBz(d_bz);
90  fitter.setPropagateToPCA(true);
91  fitter.setMaxR(200.);
92  fitter.setMinParamChange(1e-3);
93  fitter.setMinRelChi2Change(0.9);
94  fitter.setMaxDZIni(1e9);
95  fitter.setMaxChi2(1e9);
96  fitter.setUseAbsDCA(d_UseAbsDCA);
97 
98  Long_t lNCand = 0;
99 
100  std::array<float, 3> pVtx = {collision.posX(), collision.posY(), collision.posZ()};
101 
102  for (auto& t0 : goodPosTracks) { //FIXME: turn into combination(...)
103  if (t0.tpcNClsCrossedRows() < 70)
104  continue; //FIXME: turn into extra filter
105  for (auto& t1 : goodNegTracks) {
106  if (t1.tpcNClsCrossedRows() < 70)
107  continue; //FIXME: turn into extra filter
108 
109  auto Track1 = getTrackParCov(t0);
110  auto Track2 = getTrackParCov(t1);
111 
112  //Try to progate to dca
113  int nCand = fitter.process(Track1, Track2);
114  if (nCand == 0)
115  continue;
116  const auto& vtx = fitter.getPCACandidate();
117 
118  //Fiducial: min radius
119  auto thisv0radius = TMath::Sqrt(TMath::Power(vtx[0], 2) + TMath::Power(vtx[1], 2));
120  if (thisv0radius < v0radius)
121  continue;
122 
123  //DCA V0 daughters
124  auto thisdcav0dau = fitter.getChi2AtPCACandidate();
125  if (thisdcav0dau > dcav0dau)
126  continue;
127 
128  std::array<float, 3> pos = {0.};
129  std::array<float, 3> pvec0;
130  std::array<float, 3> pvec1;
131  for (int i = 0; i < 3; i++)
132  pos[i] = vtx[i];
133  fitter.getTrack(0).getPxPyPzGlo(pvec0);
134  fitter.getTrack(1).getPxPyPzGlo(pvec1);
135 
136  auto thisv0cospa = RecoDecay::CPA(array{collision.posX(), collision.posY(), collision.posZ()},
137  array{vtx[0], vtx[1], vtx[2]}, array{pvec0[0] + pvec1[0], pvec0[1] + pvec1[1], pvec0[2] + pvec1[2]});
138  if (thisv0cospa < v0cospa)
139  continue;
140 
141  lNCand++;
142  v0finderdata(t0.globalIndex(), t1.globalIndex(), t0.collisionId());
143  v0data(pos[0], pos[1], pos[2],
144  pvec0[0], pvec0[1], pvec0[2],
145  pvec1[0], pvec1[1], pvec1[2],
146  fitter.getChi2AtPCACandidate(),
147  t0.dcaXY(), -t1.dcaXY());
148  }
149  }
150  hCandPerEvent->Fill(lNCand);
151  }
152 };

181  void process(soa::Join<aod::Collisions, aod::EvSels, aod::Cents>::iterator const& collision, aod::FullTracks const& tracks,
182  soa::Filtered<soa::Join<aod::V0FinderData, aod::V0DataExt>> const& fullV0s)
183  {
184  if (!collision.alias()[kINT7])
185  return;
186  if (!collision.sel7())
187  return;
188 
189  Long_t lNCand = 0;
190  for (auto& v0 : fullV0s) {
191  if (v0.v0radius() > v0radius && v0.v0cosPA(collision.posX(), collision.posY(), collision.posZ()) > v0cospa) {
192  hV0Radius->Fill(v0.v0radius());
193  hV0CosPA->Fill(v0.v0cosPA(collision.posX(), collision.posY(), collision.posZ()));
194  hDCAPosToPV->Fill(v0.dcapostopv());
195  hDCANegToPV->Fill(v0.dcanegtopv());
196  hDCAV0Dau->Fill(v0.dcaV0daughters());
197 
198  hTest->Fill(v0.posTrack().pt());
199 
200  if (TMath::Abs(v0.yLambda()) < 0.5) {
201  h3dMassLambda->Fill(collision.centV0M(), v0.pt(), v0.mLambda());
202  h3dMassAntiLambda->Fill(collision.centV0M(), v0.pt(), v0.mAntiLambda());
203  }
204  if (TMath::Abs(v0.yK0Short()) < 0.5)
205  h3dMassK0Short->Fill(collision.centV0M(), v0.pt(), v0.mK0Short());
206  lNCand++;
207  }
208  }
209  hCandPerEvent->Fill(lNCand);
210  }
