93 // Creates the sink for FairLogger / InfoLogger integration
94 auto createInfoLoggerSinkHelper(InfoLogger* logger, InfoLoggerContext* ctx)
95 {
96  return [logger,
97  ctx](const std::string& content, const fair::LogMetaData& metadata) {
98  // translate FMQ metadata
99  InfoLogger::InfoLogger::Severity severity = InfoLogger::Severity::Undefined;
100  int level = InfoLogger::undefinedMessageOption.level;
101 
102  if (metadata.severity_name == fair::Logger::SeverityName(fair::Severity::nolog)) {
103  // discard
104  return;
105  } else if (metadata.severity_name == fair::Logger::SeverityName(fair::Severity::fatal)) {
106  severity = InfoLogger::Severity::Fatal;
107  } else if (metadata.severity_name == fair::Logger::SeverityName(fair::Severity::error)) {
108  severity = InfoLogger::Severity::Error;
109  } else if (metadata.severity_name == fair::Logger::SeverityName(fair::Severity::warn)) {
110  severity = InfoLogger::Severity::Warning;
111  } else if (metadata.severity_name == fair::Logger::SeverityName(fair::Severity::state)) {
112  severity = InfoLogger::Severity::Info;
113  level = 10;
114  } else if (metadata.severity_name == fair::Logger::SeverityName(fair::Severity::info)) {
115  severity = InfoLogger::Severity::Info;
116  } else if (metadata.severity_name == fair::Logger::SeverityName(fair::Severity::debug)) {
117  severity = InfoLogger::Severity::Debug;
118  } else if (metadata.severity_name == fair::Logger::SeverityName(fair::Severity::debug1)) {
119  severity = InfoLogger::Severity::Debug;
120  level = 10;
121  } else if (metadata.severity_name == fair::Logger::SeverityName(fair::Severity::debug2)) {
122  severity = InfoLogger::Severity::Debug;
123  level = 20;
124  } else if (metadata.severity_name == fair::Logger::SeverityName(fair::Severity::debug3)) {
125  severity = InfoLogger::Severity::Debug;
126  level = 30;
127  } else if (metadata.severity_name == fair::Logger::SeverityName(fair::Severity::debug4)) {
128  severity = InfoLogger::Severity::Debug;
129  level = 40;
130  } else if (metadata.severity_name == fair::Logger::SeverityName(fair::Severity::trace)) {
131  severity = InfoLogger::Severity::Debug;
132  level = 50;
133  }
134 
135  InfoLogger::InfoLoggerMessageOption opt = {
136  severity,
137  level,
138  InfoLogger::undefinedMessageOption.errorCode,
139  metadata.file.c_str(),
140  atoi(metadata.line.c_str())};
141 
142  if (logger) {
143  logger->log(opt, *ctx, "DPL: %s", content.c_str());
144  }
145  };
146 };

392 auto sendRelayerMetrics(ServiceRegistry& registry, DataProcessingStats& stats) -> void
393 {
394  if (stats.beginIterationTimestamp - stats.lastSlowMetricSentTimestamp < 5000) {
395  return;
396  }
397  ZoneScopedN("send metrics");
398  auto& relayerStats = registry.get<DataRelayer>().getStats();
399  auto& monitoring = registry.get<Monitoring>();
400 
401  O2_SIGNPOST_START(MonitoringStatus::ID, MonitoringStatus::SEND, 0, 0, O2_SIGNPOST_BLUE);
402 
403  monitoring.send(Metric{(int)relayerStats.malformedInputs, "malformed_inputs"}.addTag(Key::Subsystem, Value::DPL));
404  monitoring.send(Metric{(int)relayerStats.droppedComputations, "dropped_computations"}.addTag(Key::Subsystem, Value::DPL));
405  monitoring.send(Metric{(int)relayerStats.droppedIncomingMessages, "dropped_incoming_messages"}.addTag(Key::Subsystem, Value::DPL));
406  monitoring.send(Metric{(int)relayerStats.relayedMessages, "relayed_messages"}.addTag(Key::Subsystem, Value::DPL));
407 
408  monitoring.send(Metric{(int)stats.pendingInputs, "inputs/relayed/pending"}.addTag(Key::Subsystem, Value::DPL));
409  monitoring.send(Metric{(int)stats.incomplete, "inputs/relayed/incomplete"}.addTag(Key::Subsystem, Value::DPL));
410  monitoring.send(Metric{(int)stats.inputParts, "inputs/relayed/total"}.addTag(Key::Subsystem, Value::DPL));
411  monitoring.send(Metric{stats.lastElapsedTimeMs, "elapsed_time_ms"}.addTag(Key::Subsystem, Value::DPL));
412  monitoring.send(Metric{stats.lastTotalProcessedSize, "processed_input_size_byte"}
413  .addTag(Key::Subsystem, Value::DPL));
414  monitoring.send(Metric{(stats.lastTotalProcessedSize.load() / (stats.lastElapsedTimeMs.load() ? stats.lastElapsedTimeMs.load() : 1) / 1000),
415  "processing_rate_mb_s"}
416  .addTag(Key::Subsystem, Value::DPL));
417  monitoring.send(Metric{stats.lastLatency.minLatency, "min_input_latency_ms"}
418  .addTag(Key::Subsystem, Value::DPL));
419  monitoring.send(Metric{stats.lastLatency.maxLatency, "max_input_latency_ms"}
420  .addTag(Key::Subsystem, Value::DPL));
421  monitoring.send(Metric{(stats.lastTotalProcessedSize / (stats.lastLatency.maxLatency ? stats.lastLatency.maxLatency : 1) / 1000), "input_rate_mb_s"}
422  .addTag(Key::Subsystem, Value::DPL));
423 
424  stats.lastSlowMetricSentTimestamp.store(stats.beginIterationTimestamp.load());
425  O2_SIGNPOST_END(MonitoringStatus::ID, MonitoringStatus::SEND, 0, 0, O2_SIGNPOST_BLUE);
426 };

429 auto flushMetrics(ServiceRegistry& registry, DataProcessingStats& stats) -> void
430 {
431  if (stats.beginIterationTimestamp - stats.lastMetricFlushedTimestamp < 1000) {
432  return;
433  }
434  ZoneScopedN("flush metrics");
435  auto& monitoring = registry.get<Monitoring>();
436  auto& relayer = registry.get<DataRelayer>();
437 
438  O2_SIGNPOST_START(MonitoringStatus::ID, MonitoringStatus::FLUSH, 0, 0, O2_SIGNPOST_RED);
439  // Send all the relevant metrics for the relayer to update the GUI
440  // FIXME: do a delta with the previous version if too many metrics are still
441  // sent...
442  for (size_t si = 0; si < stats.statesSize.load(); ++si) {
443  auto value = std::atomic_load_explicit(&stats.relayerState[si], std::memory_order_relaxed);
444  std::atomic_thread_fence(std::memory_order_acquire);
445  monitoring.send({value, fmt::format("data_relayer/{}", si)});
446  }
447  relayer.sendContextState();
448  monitoring.flushBuffer();
449  stats.lastMetricFlushedTimestamp.store(stats.beginIterationTimestamp.load());
450  O2_SIGNPOST_END(MonitoringStatus::ID, MonitoringStatus::FLUSH, 0, 0, O2_SIGNPOST_RED);
451 };
