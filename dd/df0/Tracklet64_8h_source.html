64  // ----- Getters for contents of tracklet word -----
65  uint64_t getHCID() const { return ((mtrackletWord & hcidmask) >> hcidbs); }; // no units 0..1077
66  uint64_t getPadRow() const { return ((mtrackletWord & padrowmask) >> padrowbs); }; // in units of
67  uint64_t getColumn() const { return ((mtrackletWord & colmask) >> colbs); }; // in units of
68  uint64_t getPosition() const { return ((mtrackletWord & posmask) >> posbs); }; // in units of 0.02 pads [10bits] .. -10.22 to 10.22
69  uint64_t getSlope() const { return ((mtrackletWord & slopemask) >> slopebs); }; // in units of -127 .. 127
70  uint64_t getPID() const { return ((mtrackletWord & PIDmask)); }; // in units of counts all 3 together
71  uint64_t getQ0() const { return ((mtrackletWord & Q0mask) >> Q0bs); }; // in units of counts all 3 together
72  uint64_t getQ1() const { return ((mtrackletWord & Q1mask) >> Q1bs); }; // in units of counts all 3 together
73  uint64_t getQ2() const { return ((mtrackletWord & Q2mask) >> Q2bs); }; // in units of counts all 3 together

75  uint64_t buildTrackletWord(uint64_t format, uint64_t hcid, uint64_t padrow, uint64_t col, uint64_t position, uint64_t slope, uint64_t Q2, uint64_t Q1, uint64_t Q0)
76  {
77  mtrackletWord = ((format << formatbs) & formatmask) + ((hcid << hcidbs) & hcidmask) + ((padrow << padrowbs) & padrowmask) + ((col << colbs) & colmask) + ((position << posbs) & posmask) + ((slope << slopebs) & slopemask) + ((Q2 << Q2bs) & Q2mask) + ((Q1 << Q1bs) & Q1mask) + ((Q0 << Q0bs) & Q0mask);
78  return 0;
79  }
80  uint64_t setTrackletWord(uint64_t trackletword)
81  {
82  mtrackletWord = trackletword;
83  return 0;
84  }

86  // ----- Getters for tracklet information -----
87  int getMCM() const
88  {
89  return (getColumn() % (72)) / 18 + 4 * (getPadRow() % 4);
90  }
91  int getROB() const
92  {
93  int side = getColumn() / 72;
94  return (int)((int)getPadRow() / 8 + side);
95  }
