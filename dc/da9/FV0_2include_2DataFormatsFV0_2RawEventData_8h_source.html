131  std::vector<char> to_vector(bool tcm)
132  {
133  constexpr int CRUWordSize = 16;
134 
135  std::vector<char> result(size() * CRUWordSize);
136  char* out = result.data();
137  if (!tcm) {
138  std::memcpy(out, &mEventHeader, EventHeader::PayloadSize);
139  out += EventHeader::PayloadSize;
140  LOG(DEBUG) << " Write PM header: nWords: " << (int)mEventHeader.nGBTWords
141  << " orbit: " << int(mEventHeader.orbit)
142  << " BC: " << int(mEventHeader.bc)
143  << " size: " << result.size();
144  printHexEventHeader();
145  out += CRUWordSize - EventHeader::PayloadSize; // Padding enabled
146 
147  for (uint64_t i = 0; i < mEventHeader.nGBTWords; ++i) {
148  std::memcpy(out, &mEventData[2 * i], EventData::PayloadSizeFirstWord);
149  out += EventData::PayloadSizeFirstWord;
150  LOG(DEBUG) << " 1st word: Ch: " << std::setw(2) << mEventData[2 * i].channelID
151  << " charge: " << std::setw(4) << mEventData[2 * i].charge
152  << " time: " << std::setw(4) << mEventData[2 * i].time;
153  std::memcpy(out, &mEventData[2 * i + 1], EventData::PayloadSizeSecondWord);
154  out += EventData::PayloadSizeSecondWord;
155  LOG(DEBUG) << " 2nd word: Ch: " << std::setw(2) << mEventData[2 * i + 1].channelID
156  << " charge: " << std::setw(4) << mEventData[2 * i + 1].charge
157  << " time: " << std::setw(4) << mEventData[2 * i + 1].time;
158 
159  out += CRUWordSize - EventData::PayloadSizeSecondWord - EventData::PayloadSizeFirstWord;
160  printHexEventData(i);
161  }
162  } else {
163  // TCM data
164  std::memcpy(out, &mEventHeader, EventHeader::PayloadSize);
165  out += EventHeader::PayloadSize;
166  LOG(DEBUG) << " Write TCM header: nWords: " << (int)mEventHeader.nGBTWords
167  << " orbit: " << int(mEventHeader.orbit)
168  << " BC: " << int(mEventHeader.bc)
169  << " size: " << result.size();
170  std::memcpy(out, &mTCMdata, sizeof(TCMdata));
171  out += sizeof(TCMdata);
172  // TODO: No TCM payload printing until the meaning of trigger bits and other flags is clarified
173  }
174  return result;
175  }

180  TCMdata mTCMdata;
