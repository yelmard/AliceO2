18 // This task shows how to access the ZDC and the Run2 V0 information which belongs to a collision
19 // The association is made through the BC column (and in Run 3 may not be unique!)
20 // To run this workflow, the o2-analysis-run2-matcher has to be run as well.
21 // Example: o2-analysis-run2-matcher --aod-file AO2D.root | o2-analysistutorial-zdc-vzero-iteration
22 
23 // This example access the collisions and the related Run2V0 information.
24 // Note that one has to subscribe to aod::Collisions const& and aod::Run2V0s const& to load
25 // the relevant data even if you access the data itself through m.collision() and m.run2v0()
26 // Here the "sparse" matcher is used which means, there can be collisions without Run2V0 information
27 // To find out, m.has_run2v0() has to be called. Otherwise m.run2v0() will fail.
28 struct IterateV0 {
29  void process(aod::Run2MatchedSparse::iterator const& m, aod::Collisions const&, aod::Run2V0s const&)
30  {
31  LOGF(INFO, "Vertex = %f", m.collision().posZ());
32  if (m.has_run2v0()) {
33  auto v0 = m.run2v0();
34  LOGF(info, "V0: %f %f", v0.adc()[0], v0.adc()[1]);
35  } else {
36  LOGF(INFO, "No V0 info");
37  }
38  }
39 };

41 // This example is identical to IterateV0, but uses the exclusive match. This means that lines where any

53 // This example builds on IterateV0 and in addition accesses also the tracks grouped to the specific collision.
