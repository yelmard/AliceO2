109 template <typename coder_T, typename stream_T, typename source_T>
110 template <typename stream_IT, typename source_IT>
111 void Decoder<coder_T, stream_T, source_T>::process(const source_IT outputBegin, const stream_IT inputEnd, size_t messageLength) const
112 {
113  using namespace internal;
114  LOG(trace) << "start decoding";
115  RANSTimer t;
116  t.start();
117  static_assert(std::is_same<typename std::iterator_traits<source_IT>::value_type, source_T>::value);
118  static_assert(std::is_same<typename std::iterator_traits<stream_IT>::value_type, stream_T>::value);
119 
120  if (messageLength == 0) {
121  LOG(warning) << "Empty message passed to decoder, skipping decode process";
122  return;
123  }
124 
125  stream_IT inputIter = inputEnd;
126  source_IT it = outputBegin;
127 
128  // make Iter point to the last last element
129  --inputIter;
130 
131  ransDecoder rans0, rans1;
132  inputIter = rans0.init(inputIter);
133  inputIter = rans1.init(inputIter);
134 
135  for (size_t i = 0; i < (messageLength & ~1); i += 2) {
136  const int64_t s0 = (*mReverseLUT)[rans0.get(mProbabilityBits)];
137  const int64_t s1 = (*mReverseLUT)[rans1.get(mProbabilityBits)];
138  *it++ = s0;
139  *it++ = s1;
140  inputIter = rans0.advanceSymbol(inputIter, (*mSymbolTable)[s0], mProbabilityBits);
141  inputIter = rans1.advanceSymbol(inputIter, (*mSymbolTable)[s1], mProbabilityBits);
142  }
143 
144  // last byte, if message length was odd
145  if (messageLength & 1) {
146  const int64_t s0 = (*mReverseLUT)[rans0.get(mProbabilityBits)];
147  *it = s0;
148  inputIter = rans0.advanceSymbol(inputIter, (*mSymbolTable)[s0], mProbabilityBits);
149  }
150  t.stop();
151  LOG(debug1) << "Decoder::" << __func__ << " { DecodedSymbols: " << messageLength << ","
152  << "processedBytes: " << messageLength * sizeof(source_T) << ","
153  << " inclusiveTimeMS: " << t.getDurationMS() << ","
154  << " BandwidthMiBPS: " << std::fixed << std::setprecision(2) << (messageLength * sizeof(source_T) * 1.0) / (t.getDurationS() * 1.0 * (1 << 20)) << "}";
155 
156  LOG(trace) << "done decoding";
157 }
