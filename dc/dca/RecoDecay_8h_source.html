152  {
153  // eta = arctanh(pz/p)
154  if (std::abs(mom[0]) < Almost0 && std::abs(mom[1]) < Almost0) { // very small px and py

176  template <typename T, typename U, typename V>
177  static auto CPA(const array<T, 3>& posPV, const array<U, 3>& posSV, const array<V, 3>& mom)
178  {
179  // CPA = (l . p)/(|l| |p|)
180  auto lineDecay = array{posSV[0] - posPV[0], posSV[1] - posPV[1], posSV[2] - posPV[2]};
181  auto cos = dotProd(lineDecay, mom) / std::sqrt(mag2(lineDecay) * mag2(mom));
182  if (cos < -1.) {
183  return -1.;
184  }
185  if (cos > 1.) {
186  return 1.;
187  }
188  return cos;
189  }

196  template <std::size_t N, std::size_t O, std::size_t P, typename T, typename U, typename V>
197  static auto CPAXY(const array<T, N>& posPV, const array<U, O>& posSV, const array<V, P>& mom)
198  {
199  // CPAXY = (r . pT)/(|r| |pT|)
200  auto lineDecay = array{posSV[0] - posPV[0], posSV[1] - posPV[1]};
201  auto momXY = array{mom[0], mom[1]};
202  auto cos = dotProd(lineDecay, momXY) / std::sqrt(mag2(lineDecay) * mag2(momXY));
203  if (cos < -1.) {
204  return -1.;
205  }
206  if (cos > 1.) {
207  return 1.;
208  }
209  return cos;
210  }

232  template <typename T, typename U>
233  static auto CosThetaStarM(const array<array<T, 3>, 2>& arrMom, const array<U, 2>& arrMass, int iProng)
234  {
235  auto eProng = E(arrMom[iProng], arrMass[iProng]); // energy of the prong
236  auto pVecTot = PVec(arrMom[0], arrMom[1]); // momentum of the 2-prong system
237  auto pTot = P(pVecTot); // magnitude of the momentum of the 2-prong system
238  auto mTot = M(arrMom, arrMass); // invariant mass of the 2-prong system
239  auto eTot = eProng + E(arrMom[1 - iProng], arrMass[1 - iProng]); // energy of the 2-prong system
240  auto gamma = eTot / mTot; // γ, Lorentz gamma factor of the 2-prong system
241  auto beta = pTot / eTot; // β, velocity of the 2-prong system
242  auto pStar = std::sqrt(sq(sq(mTot) - sq(arrMass[0]) - sq(arrMass[1])) - sq(2 * arrMass[0] * arrMass[1])) / (2 * mTot); // p*, prong momentum in the rest frame of the 2-prong system
243  // p* = √[(M^2 - m1^2 - m2^2)^2 - 4 m1^2 m2^2]/2M
244  // Lorentz transformation of the longitudinal momentum of the prong into the rest frame of the 2-prong system:
245  // p*_L,i = γ (p_L,i - β E_i)
246  // p*_L,i = p* cos(θ*_i)
247  // cos(θ*_i) = γ (p_L,i - β E_i)/p*
248  return gamma * (dotProd(arrMom[iProng], pVecTot) / pTot - beta * eProng) / pStar;
249  }

258  template <typename T, typename U>
259  static auto CosThetaStarD(const array<array<T, 3>, 2>& arrMom, const array<U, 2>& arrMass, int iProng)
260  {
261  auto eProng = E(arrMom[iProng], arrMass[iProng]); // energy of the prong
262  auto pVecTot = PVec(arrMom[0], arrMom[1]); // momentum of the 2-prong system
263  auto pTot = P(pVecTot); // magnitude of the momentum of the 2-prong system
264  auto mTot = M(arrMom, arrMass); // invariant mass of the 2-prong system
265  auto eTot = eProng + E(arrMom[1 - iProng], arrMass[1 - iProng]); // energy of the 2-prong system
266  auto gamma = eTot / mTot; // γ, Lorentz gamma factor of the 2-prong system
267  auto beta = pTot / eTot; // β, velocity of the 2-prong system
268  auto pStar = std::sqrt(sq(sq(mTot) - sq(arrMass[0]) - sq(arrMass[1])) - sq(2 * arrMass[0] * arrMass[1])) / (2 * mTot); // p*, prong momentum in the rest frame of the 2-prong system
269  // p* = √[(M^2 - m1^2 - m2^2)^2 - 4 m1^2 m2^2]/2M
270  // Lorentz transformation of the longitudinal momentum of the prong into the detector frame:
271  // p_L,i = γ (p*_L,i + β E*_i)
272  // p*_L,i = p_L,i/γ - β E*_i (less sensitive too incorrect masses?)
273  // cos(θ*_i) = (p_L,i/γ - β E*_i)/p*
274  return (dotProd(arrMom[iProng], pVecTot) / (pTot * gamma) - beta * E(pStar, arrMass[iProng])) / pStar;
275  }

285  template <typename T, typename U, typename V>
286  static auto CosThetaStarA(const array<array<T, 3>, 2>& arrMom, const array<U, 2>& arrMass, V mTot, int iProng)
287  {
288  auto pVecTot = PVec(arrMom[0], arrMom[1]); // momentum of the mother particle
289  auto pTot = P(pVecTot); // magnitude of the momentum of the mother particle
290  auto eTot = E(pTot, mTot); // energy of the mother particle
291  auto gamma = eTot / mTot; // γ, Lorentz gamma factor of the mother particle
292  auto beta = pTot / eTot; // β, velocity of the mother particle
293  auto pStar = std::sqrt(sq(sq(mTot) - sq(arrMass[0]) - sq(arrMass[1])) - sq(2 * arrMass[0] * arrMass[1])) / (2 * mTot); // p*, prong momentum in the rest frame of the mother particle
294  // p* = √[(M^2 - m1^2 - m2^2)^2 - 4 m1^2 m2^2]/2M
295  // Lorentz transformation of the longitudinal momentum of the prong into the detector frame:
296  // p_L,i = γ (p*_L,i + β E*_i)
297  // p*_L,i = p_L,i/γ - β E*_i
298  // cos(θ*_i) = (p_L,i/γ - β E*_i)/p*
299  return (dotProd(arrMom[iProng], pVecTot) / (pTot * gamma) - beta * E(pStar, arrMass[iProng])) / pStar;
300  }

418  template <std::size_t N, typename T, typename U>
419  static auto M2(const array<array<T, 3>, N>& arrMom, const array<U, N>& arrMass)
420  {
421  array<double, 3> momTotal{0., 0., 0.}; // candidate momentum vector
422  double energyTot{0.}; // candidate energy
423  for (auto iProng = 0; iProng < N; ++iProng) {
424  for (auto iMom = 0; iMom < 3; ++iMom) {
425  momTotal[iMom] += arrMom[iProng][iMom];
426  } // loop over momentum components
427  energyTot += E(arrMom[iProng], arrMass[iProng]);
428  } // loop over prongs
429  return energyTot * energyTot - P2(momTotal);
430  }
