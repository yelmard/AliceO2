59  static size_t alignSize(T*& var, size_t n = 1)
60  {
61  auto sz = sizeof(T) * n;

108 template <int NU, int NL, typename CU, typename CL>
109 void CTFCoder::splitColumns(const std::vector<MTYPE(NU, NL)>& vm, CU*& vu, CL*& vl)
110 {
111  static_assert(NU <= sizeof(CU) * 8 && NL <= sizeof(CL) * 8, "output columns bit count is wrong");
112  size_t n = vm.size();
113  for (size_t i = 0; i < n; i++) {
114  vu[i] = static_cast<CU>(vm[i] >> NL);
115  vl[i] = static_cast<CL>(vm[i] & ((0x1 << NL) - 1));
116  }
117 }
118 
120 template <int NU, int NL, typename CU, typename CL>
121 auto CTFCoder::mergeColumns(const CU* vu, const CL* vl, size_t nelem)
122 {
123  // merge 2 columns to 1
124  static_assert(NU <= sizeof(NU) * 8 && NL <= sizeof(NL) * 8, "input columns bit count is wrong");
125  std::vector<MTYPE(NU, NL)> outv;
126  outv.reserve(nelem);
127  for (size_t i = 0; i < nelem; i++) {
128  outv.push_back((static_cast<MTYPE(NU, NL)>(vu[i]) << NL) | static_cast<MTYPE(NU, NL)>(vl[i]));
129  }
130  return std::move(outv);
131 }

134 template <typename VEC>
135 void CTFCoder::encode(VEC& buff, const CompressedClusters& ccl)
136 {
137  using MD = o2::ctf::Metadata::OptStore;
138  // what to do which each field: see o2::ctf::Metadata explanation
139  constexpr MD optField[CTF::getNBlocks()] = {
140  MD::EENCODE, //qTotA
141  MD::EENCODE, //qMaxA
142  MD::EENCODE, //flagsA
143  MD::EENCODE, //rowDiffA
144  MD::EENCODE, //sliceLegDiffA
145  MD::EENCODE, //padResA
146  MD::EENCODE, //timeResA
147  MD::EENCODE, //sigmaPadA
148  MD::EENCODE, //sigmaTimeA
149  MD::EENCODE, //qPtA
150  MD::EENCODE, //rowA
151  MD::EENCODE, //sliceA
152  MD::EENCODE, //timeA
153  MD::EENCODE, //padA
154  MD::EENCODE, //qTotU
155  MD::EENCODE, //qMaxU
156  MD::EENCODE, //flagsU
157  MD::EENCODE, //padDiffU
158  MD::EENCODE, //timeDiffU
159  MD::EENCODE, //sigmaPadU
160  MD::EENCODE, //sigmaTimeU
161  MD::EENCODE, //nTrackClusters
162  MD::EENCODE //nSliceRowClusters
163  };
164 
165  // book output size with some margin
166  auto szIni = estimateCompressedSize(ccl);
167  buff.resize(szIni);
168 
169  auto ec = CTF::create(buff);
170  uint32_t flags = 0;
171  if (mCombineColumns) {
172  flags |= CTFHeader::CombinedColumns;
173  }
174  ec->setHeader(CTFHeader{reinterpret_cast<const CompressedClustersCounters&>(ccl), flags});
175  ec->getANSHeader().majorVersion = 0;
176  ec->getANSHeader().minorVersion = 1;
177  // at every encoding the buffer might be autoexpanded, so we don't work with fixed pointer ec
178 #define ENCODETPC(beg, end, slot, bits) CTF::get(buff.data())->encode(beg, end, int(slot), bits, optField[int(slot)], &buff, mCoders[int(slot)].get());
179  // clang-format off
180 
181  if (mCombineColumns) {
182  auto mrg = mergeColumns<CTF::NBitsQTot, CTF::NBitsQMax>(ccl.qTotA, ccl.qMaxA, ccl.nAttachedClusters);
183  ENCODETPC(&mrg[0], (&mrg[0]) + ccl.nAttachedClusters, CTF::BLCqTotA, 0);
184  }
185  else {
186  ENCODETPC(ccl.qTotA, ccl.qTotA + ccl.nAttachedClusters, CTF::BLCqTotA, 0);
187  }
188  ENCODETPC(ccl.qMaxA, ccl.qMaxA + (mCombineColumns ? 0 : ccl.nAttachedClusters), CTF::BLCqMaxA, 0);
189 
190  ENCODETPC(ccl.flagsA, ccl.flagsA + ccl.nAttachedClusters, CTF::BLCflagsA, 0);
191 
192  if (mCombineColumns) {
193  auto mrg = mergeColumns<CTF::NBitsRowDiff, CTF::NBitsSliceLegDiff>(ccl.rowDiffA, ccl.sliceLegDiffA, ccl.nAttachedClustersReduced);
194  ENCODETPC(&mrg[0], (&mrg[0]) + ccl.nAttachedClustersReduced, CTF::BLCrowDiffA, 0);
195  }
196  else {
197  ENCODETPC(ccl.rowDiffA, ccl.rowDiffA + ccl.nAttachedClustersReduced, CTF::BLCrowDiffA, 0);
198  }
199  ENCODETPC(ccl.sliceLegDiffA, ccl.sliceLegDiffA + (mCombineColumns ? 0 : ccl.nAttachedClustersReduced), CTF::BLCsliceLegDiffA, 0);
200 
201  ENCODETPC(ccl.padResA, ccl.padResA + ccl.nAttachedClustersReduced, CTF::BLCpadResA, 0);
202  ENCODETPC(ccl.timeResA, ccl.timeResA + ccl.nAttachedClustersReduced, CTF::BLCtimeResA, 0);
203 
204  if (mCombineColumns) {
205  auto mrg = mergeColumns<CTF::NBitsSigmaPad, CTF::NBitsSigmaTime>(ccl.sigmaPadA, ccl.sigmaTimeA, ccl.nAttachedClusters);
206  ENCODETPC(&mrg[0], &mrg[0] + ccl.nAttachedClusters, CTF::BLCsigmaPadA, 0);
207  }
208  else {
209  ENCODETPC(ccl.sigmaPadA, ccl.sigmaPadA + ccl.nAttachedClusters, CTF::BLCsigmaPadA, 0);
210  }
211  ENCODETPC(ccl.sigmaTimeA, ccl.sigmaTimeA + (mCombineColumns ? 0 : ccl.nAttachedClusters), CTF::BLCsigmaTimeA, 0);
212 
213  ENCODETPC(ccl.qPtA, ccl.qPtA + ccl.nTracks, CTF::BLCqPtA, 0);
214  ENCODETPC(ccl.rowA, ccl.rowA + ccl.nTracks, CTF::BLCrowA, 0);
215  ENCODETPC(ccl.sliceA, ccl.sliceA + ccl.nTracks, CTF::BLCsliceA, 0);
216  ENCODETPC(ccl.timeA, ccl.timeA + ccl.nTracks, CTF::BLCtimeA, 0);
217  ENCODETPC(ccl.padA, ccl.padA + ccl.nTracks, CTF::BLCpadA, 0);
218 
219  if (mCombineColumns) {
220  auto mrg = mergeColumns<CTF::NBitsQTot, CTF::NBitsQMax>(ccl.qTotU,ccl.qMaxU,ccl.nUnattachedClusters);
221  ENCODETPC(&mrg[0], &mrg[0] + ccl.nUnattachedClusters, CTF::BLCqTotU, 0);
222  }
223  else {
224  ENCODETPC(ccl.qTotU, ccl.qTotU + ccl.nUnattachedClusters, CTF::BLCqTotU, 0);
225  }
226  ENCODETPC(ccl.qMaxU, ccl.qMaxU + (mCombineColumns ? 0 : ccl.nUnattachedClusters), CTF::BLCqMaxU, 0);
227 
228  ENCODETPC(ccl.flagsU, ccl.flagsU + ccl.nUnattachedClusters, CTF::BLCflagsU, 0);
229  ENCODETPC(ccl.padDiffU, ccl.padDiffU + ccl.nUnattachedClusters, CTF::BLCpadDiffU, 0);
230  ENCODETPC(ccl.timeDiffU, ccl.timeDiffU + ccl.nUnattachedClusters, CTF::BLCtimeDiffU, 0);
231 
232  if (mCombineColumns) {
233  auto mrg = mergeColumns<CTF::NBitsSigmaPad, CTF::NBitsSigmaTime>(ccl.sigmaPadU, ccl.sigmaTimeU, ccl.nUnattachedClusters);
234  ENCODETPC(&mrg[0], &mrg[0] + ccl.nUnattachedClusters, CTF::BLCsigmaPadU, 0);
235  }
236  else {
237  ENCODETPC(ccl.sigmaPadU, ccl.sigmaPadU + ccl.nUnattachedClusters, CTF::BLCsigmaPadU, 0);
238  }
239  ENCODETPC(ccl.sigmaTimeU, ccl.sigmaTimeU + (mCombineColumns ? 0 : ccl.nUnattachedClusters), CTF::BLCsigmaTimeU, 0);
240 
241  ENCODETPC(ccl.nTrackClusters, ccl.nTrackClusters + ccl.nTracks, CTF::BLCnTrackClusters, 0);
242  ENCODETPC(ccl.nSliceRowClusters, ccl.nSliceRowClusters + ccl.nSliceRows, CTF::BLCnSliceRowClusters, 0);
243  // clang-format on
244  CTF::get(buff.data())->print(getPrefix());
245 }

248 template <typename VEC>
249 void CTFCoder::decode(const CTF::base& ec, VEC& buffVec)
250 {
251  CompressedClusters cc;
252  CompressedClustersCounters& ccCount = cc;
253  auto& header = ec.getHeader();
254  checkDataDictionaryConsistency(header);
255  ccCount = reinterpret_cast<const CompressedClustersCounters&>(header);
256  CompressedClustersFlat* ccFlat = nullptr;
257  size_t sizeCFlatBody = alignSize(ccFlat);
258  size_t sz = sizeCFlatBody + estimateSize(cc); // total size of the buffVec accounting for the alignment
259  size_t vsz = sizeof(typename std::remove_reference<decltype(buffVec)>::type::value_type); // size of the element of the buffer
260  buffVec.resize(sz / vsz);
261  ccFlat = reinterpret_cast<CompressedClustersFlat*>(buffVec.data()); // RS? do we need to align this pointer, or PMR vector will be already aligned?
262  auto buff = reinterpret_cast<void*>(reinterpret_cast<char*>(buffVec.data()) + sizeCFlatBody); // will be the start of the CompressedClustersFlat payload
263 
264  setCompClusAddresses(cc, buff);
265  ccFlat->set(sz, cc); // set offsets
266  ec.print(getPrefix());
267 
268  // decode encoded data directly to destination buff
269 #define DECODETPC(part, slot) ec.decode(part, int(slot), mCoders[int(slot)].get())
270  // clang-format off
271  if (mCombineColumns) {
272  std::vector<MTYPE(CTF::NBitsQTot, CTF::NBitsQMax)> mrg;
273  DECODETPC(mrg, CTF::BLCqTotA);
274  splitColumns<CTF::NBitsQTot, CTF::NBitsQMax>(mrg, cc.qTotA, cc.qMaxA);
275  }
276  else {
277  DECODETPC(cc.qTotA, CTF::BLCqTotA);
278  DECODETPC(cc.qMaxA, CTF::BLCqMaxA);
279  }
280 
281  DECODETPC(cc.flagsA, CTF::BLCflagsA);
282 
283  if (mCombineColumns) {
284  std::vector<MTYPE(CTF::NBitsRowDiff, CTF::NBitsSliceLegDiff)> mrg;
285  DECODETPC(mrg, CTF::BLCrowDiffA);
286  splitColumns<CTF::NBitsRowDiff, CTF::NBitsSliceLegDiff>(mrg, cc.rowDiffA, cc.sliceLegDiffA);
287  }
288  else {
289  DECODETPC(cc.rowDiffA, CTF::BLCrowDiffA);
290  DECODETPC(cc.sliceLegDiffA, CTF::BLCsliceLegDiffA);
291  }
292 
293  DECODETPC(cc.padResA, CTF::BLCpadResA);
294  DECODETPC(cc.timeResA, CTF::BLCtimeResA);
295 
296  if (mCombineColumns) {
297  std::vector<MTYPE(CTF::NBitsSigmaPad, CTF::NBitsSigmaTime)> mrg;
298  DECODETPC(mrg, CTF::BLCsigmaPadA);
299  splitColumns<CTF::NBitsSigmaPad, CTF::NBitsSigmaTime>(mrg, cc.sigmaPadA, cc.sigmaTimeA);
300  }
301  else {
302  DECODETPC(cc.sigmaPadA, CTF::BLCsigmaPadA);
303  DECODETPC(cc.sigmaTimeA, CTF::BLCsigmaTimeA);
304  }
305 
306  DECODETPC(cc.qPtA, CTF::BLCqPtA);
307  DECODETPC(cc.rowA, CTF::BLCrowA);
308  DECODETPC(cc.sliceA, CTF::BLCsliceA);
309  DECODETPC(cc.timeA, CTF::BLCtimeA);
310  DECODETPC(cc.padA, CTF::BLCpadA);
311 
312  if (mCombineColumns) {
313  std::vector<MTYPE(CTF::NBitsQTot, CTF::NBitsQMax)> mrg;
314  DECODETPC(mrg, CTF::BLCqTotU);
315  splitColumns<CTF::NBitsQTot, CTF::NBitsQMax>(mrg, cc.qTotU, cc.qMaxU);
316  }
317  else {
318  DECODETPC(cc.qTotU, CTF::BLCqTotU);
319  DECODETPC(cc.qMaxU, CTF::BLCqMaxU);
320  }
321 
322  DECODETPC(cc.flagsU, CTF::BLCflagsU);
323  DECODETPC(cc.padDiffU, CTF::BLCpadDiffU);
324  DECODETPC(cc.timeDiffU, CTF::BLCtimeDiffU);
325 
326  if (mCombineColumns) {
327  std::vector<MTYPE(CTF::NBitsSigmaPad, CTF::NBitsSigmaTime)> mrg;
328  DECODETPC(mrg, CTF::BLCsigmaPadU);
329  splitColumns<CTF::NBitsSigmaPad, CTF::NBitsSigmaTime>(mrg, cc.sigmaPadU, cc.sigmaTimeU);
330  }
331  else {
332  DECODETPC(cc.sigmaPadU, CTF::BLCsigmaPadU);
333  DECODETPC(cc.sigmaTimeU, CTF::BLCsigmaTimeU);
334  }
335 
336  DECODETPC(cc.nTrackClusters, CTF::BLCnTrackClusters);
337  DECODETPC(cc.nSliceRowClusters, CTF::BLCnSliceRowClusters);
338  // clang-format on
339 }
