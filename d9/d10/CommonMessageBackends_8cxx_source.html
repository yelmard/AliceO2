33 struct CommonMessageBackendsHelpers {
34  static ServiceInit createCallback()
35  {
36  return [](ServiceRegistry& services, DeviceState&, fair::mq::ProgOptions& options) {
37  auto& device = services.get<RawDeviceService>();
38  return ServiceHandle{TypeIdHelpers::uniqueId<T>(), new T(FairMQDeviceProxy{device.device()})};
39  };
40  }
41 
42  static ServiceProcessingCallback sendCallback()
43  {
44  return [](ProcessingContext& ctx, void* service) {
45  ZoneScopedN("send message callback");
46  T* context = reinterpret_cast<T*>(service);
47  auto& device = ctx.services().get<RawDeviceService>();

50  }

96 o2::framework::ServiceSpec CommonMessageBackends::fairMQBackendSpec()
97 {
98  return ServiceSpec{"fairmq-backend",
99  [](ServiceRegistry& services, DeviceState&, fair::mq::ProgOptions&) -> ServiceHandle {
100  auto& device = services.get<RawDeviceService>();
101  auto context = new MessageContext(FairMQDeviceProxy{device.device()});
102  auto& spec = services.get<DeviceSpec const>();
103 
104  auto dispatcher = [&device](FairMQParts&& parts, std::string const& channel, unsigned int index) {
105  DataProcessor::doSend(*device.device(), std::move(parts), channel.c_str(), index);
106  };
107 
108  auto matcher = [policy = spec.dispatchPolicy](o2::header::DataHeader const& header) {
109  if (policy.triggerMatcher == nullptr) {
110  return true;
111  }
112  return policy.triggerMatcher(Output{header});
113  };
114 
115  if (spec.dispatchPolicy.action == DispatchPolicy::DispatchOp::WhenReady) {
116  context->init(DispatchControl{dispatcher, matcher});
117  }
118  return ServiceHandle{TypeIdHelpers::uniqueId<MessageContext>(), context};
119  },
120  CommonServices::noConfiguration(),
121  CommonMessageBackendsHelpers<MessageContext>::clearContext(),
122  CommonMessageBackendsHelpers<MessageContext>::sendCallback(),
123  nullptr,
124  nullptr,
125  CommonMessageBackendsHelpers<MessageContext>::clearContextEOS(),
126  CommonMessageBackendsHelpers<MessageContext>::sendCallbackEOS(),
127  nullptr,
128  nullptr,
129  nullptr,
130  ServiceKind::Serial};
131 }
