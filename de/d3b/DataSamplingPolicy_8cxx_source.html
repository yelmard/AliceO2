41 void DataSamplingPolicy::configure(const ptree& config)
42 {
43  mName = config.get<std::string>("id");
44  if (mName.size() > 14) {
45  LOG(WARNING) << "DataSamplingPolicy name '" << mName << "' is longer than 14 characters, we have to trim it. "
46  << "Use a shorter policy name to avoid potential output name conflicts.";
47  mName.resize(14);
48  }
49 
50  auto subSpecString = config.get_optional<std::string>("subSpec").value_or("*");
51  auto subSpec = subSpecString.find_first_of("-*") != std::string::npos ? -1 : std::strtoull(subSpecString.c_str(), nullptr, 10);
52 
53  mPaths.clear();
54  size_t outputId = 0;
55  std::vector<InputSpec> inputSpecs = DataDescriptorQueryBuilder::parse(config.get<std::string>("query").c_str());
56 
57  for (const auto& inputSpec : inputSpecs) {
58 
59  if (DataSpecUtils::getOptionalSubSpec(inputSpec).has_value()) {
60  OutputSpec outputSpec{
61  {inputSpec.binding},
62  createPolicyDataOrigin(),
63  createPolicyDataDescription(mName, outputId++),
64  DataSpecUtils::getOptionalSubSpec(inputSpec).value(),
65  inputSpec.lifetime};
66 
67  mPaths.push_back({inputSpec, outputSpec});
68 
69  } else {
70  OutputSpec outputSpec{
71  {inputSpec.binding},
72  {createPolicyDataOrigin(), createPolicyDataDescription(mName, outputId++)},
73  inputSpec.lifetime};
74 
75  mPaths.push_back({inputSpec, outputSpec});
76  }
77  }
78 
79  mConditions.clear();
80  for (const auto& conditionConfig : config.get_child("samplingConditions")) {
81  mConditions.push_back(DataSamplingConditionFactory::create(conditionConfig.second.get<std::string>("condition")));
82  mConditions.back()->configure(conditionConfig.second);
83  }
84 
85  mFairMQOutputChannel = config.get_optional<std::string>("fairMQOutput").value_or("");
86 }

93 bool DataSamplingPolicy::decide(const o2::framework::DataRef& dataRef)
94 {
95  bool decision = std::all_of(mConditions.begin(), mConditions.end(),
96  [dataRef](std::unique_ptr<DataSamplingCondition>& condition) {
97  return condition->decide(dataRef);
98  });
99 
100  mTotalAcceptedMessages += decision;
101  mTotalEvaluatedMessages++;
102 
103  return decision;
104 }

106 const Output DataSamplingPolicy::prepareOutput(const ConcreteDataMatcher& input, Lifetime lifetime) const
107 {
108  auto result = mPaths.find(input);
109  if (result != mPaths.end()) {
110  auto dataType = DataSpecUtils::asConcreteDataTypeMatcher(result->second);
111  return Output{dataType.origin, dataType.description, input.subSpec, lifetime};
112  } else {
113  return Output{header::gDataOriginInvalid, header::gDataDescriptionInvalid};
114  }
115 }

154 header::DataDescription DataSamplingPolicy::createPolicyDataDescription(std::string policyName, size_t id)
155 {
156  if (id > 99) {
157  throw std::runtime_error("Maximum 100 inputs in DataSamplingPolicy are supported. Call the developers if you really need more.");
158  }
159 
160  if (policyName.size() > 14) {
161  LOG(WARNING) << "DataSamplingPolicy name '" << policyName << "' is longer than 14 characters, we have to trim it. "
162  << "Use a shorter policy name to avoid potential output name conflicts.";
163  policyName.resize(14);
164  }
165 
166  header::DataDescription outputDescription;
167  outputDescription.runtimeInit((policyName + std::to_string(id)).c_str());
168  return outputDescription;
169 }
