90 template <typename H, typename... T>
91 constexpr auto make_originals_from_type()
92 {
93  using decayed = std::decay_t<H>;
94  if constexpr (sizeof...(T) == 0) {
95  if constexpr (is_type_with_originals_v<decayed>) {
96  return typename decayed::originals{};
97  } else if constexpr (is_type_with_originals_v<typename decayed::table_t>) {
98  return typename decayed::table_t::originals{};
99  } else if constexpr (is_type_with_parent_v<decayed>) {
100  return make_originals_from_type(decayed::parent_t);
101  } else {
102  return framework::pack<decayed>{};
103  }
104  } else if constexpr (is_type_with_originals_v<decayed>) {
105  return framework::concatenate_pack(typename decayed::originals{}, make_originals_from_type<T...>());
106  } else if constexpr (is_type_with_originals_v<typename decayed::table_t>) {
107  return framework::concatenate_pack(typename decayed::table_t::originals{}, make_originals_from_type<T...>());
108  } else {
109  return framework::concatenate_pack(framework::pack<decayed>{}, make_originals_from_type<T...>());
110  }

130 template <typename T, typename ChunkingPolicy = Chunked>
131 class ColumnIterator : ChunkingPolicy
132 {
133  static constexpr char SCALE_FACTOR = std::is_same_v<std::decay_t<T>, bool> ? 3 : 0;
134 
135  public:
140  ColumnIterator(arrow::ChunkedArray const* column)
141  : mColumn{column},
142  mCurrentPos{nullptr},
143  mFirstIndex{0},
144  mCurrentChunk{0}
145  {
146  auto array = getCurrentArray();
147  mCurrent = reinterpret_cast<T const*>(array->values()->data()) + array->offset();
148  mLast = mCurrent + array->length();
149  }
150 
151  ColumnIterator() = default;
152  ColumnIterator(ColumnIterator<T, ChunkingPolicy> const&) = default;
153  ColumnIterator<T, ChunkingPolicy>& operator=(ColumnIterator<T, ChunkingPolicy> const&) = default;
154 
155  ColumnIterator(ColumnIterator<T, ChunkingPolicy>&&) = default;
156  ColumnIterator<T, ChunkingPolicy>& operator=(ColumnIterator<T, ChunkingPolicy>&&) = default;
157 
159  void nextChunk() const
160  {
161  auto previousArray = getCurrentArray();
162  mFirstIndex += previousArray->length();
163 
164  mCurrentChunk++;
165  auto array = getCurrentArray();
166  mCurrent = reinterpret_cast<T const*>(array->values()->data()) + array->offset() - (mFirstIndex >> SCALE_FACTOR);
167  mLast = mCurrent + array->length() + (mFirstIndex >> SCALE_FACTOR);
168  }
169 
170  void prevChunk() const
171  {
172  auto previousArray = getCurrentArray();
173  mFirstIndex -= previousArray->length();
174 
175  mCurrentChunk--;
176  auto array = getCurrentArray();
177  mCurrent = reinterpret_cast<T const*>(array->values()->data()) + array->offset() - (mFirstIndex >> SCALE_FACTOR);
178  mLast = mCurrent + array->length() + (mFirstIndex >> SCALE_FACTOR);
179  }
180 
181  void moveToChunk(int chunk)
182  {
183  if (mCurrentChunk < chunk) {
184  while (mCurrentChunk != chunk) {
185  nextChunk();
186  }
187  } else {
188  while (mCurrentChunk != chunk) {
189  prevChunk();
190  }
191  }
192  }
193 
195  void moveToEnd()
196  {
197  mCurrentChunk = mColumn->num_chunks() - 1;
198  auto array = getCurrentArray();
199  mFirstIndex = mColumn->length() - array->length();
200  mCurrent = reinterpret_cast<T const*>(array->values()->data()) + array->offset() - (mFirstIndex >> SCALE_FACTOR);
201  mLast = mCurrent + array->length() + (mFirstIndex >> SCALE_FACTOR);
202  }
203 
204  decltype(auto) operator*() const
205  {
206  if constexpr (ChunkingPolicy::chunked) {
207  if (O2_BUILTIN_UNLIKELY(((mCurrent + (*mCurrentPos >> SCALE_FACTOR)) >= mLast))) {
208  nextChunk();
209  }
210  }
211  if constexpr (std::is_same_v<bool, std::decay_t<T>>) {
212  // FIXME: notice that due to the type punning we cannot simply convert the
213  // masked char to a bool, because it's undefined behavior.
214  // FIXME: check if shifting the masked bit to the first position is better than != 0
215  return (*((char*)mCurrent + (*mCurrentPos >> SCALE_FACTOR)) & (1 << (*mCurrentPos & 0x7))) != 0;
216  } else {
217  return *(mCurrent + (*mCurrentPos >> SCALE_FACTOR));
218  }
219  }
220 
221  // Move to the chunk which containts element pos
222  ColumnIterator<T>& moveToPos()
223  {
224  // If we get outside range of the current chunk, go to the next.
225  if constexpr (ChunkingPolicy::chunked) {
226  while (O2_BUILTIN_UNLIKELY((mCurrent + (*mCurrentPos >> SCALE_FACTOR)) >= mLast)) {
227  nextChunk();
228  }
229  }
230  return *this;
231  }
232 
233  // Move to the chunk which containts element pos
234  ColumnIterator<T>& checkNextChunk()
235  {
236  if constexpr (ChunkingPolicy::chunked) {
237  if (O2_BUILTIN_LIKELY((mCurrent + (*mCurrentPos >> SCALE_FACTOR)) <= mLast)) {
238  return *this;
239  }
240  nextChunk();
241  }
242  return *this;
243  }
244 
245  mutable T const* mCurrent;
246  int64_t const* mCurrentPos;
247  mutable T const* mLast;
248  arrow::ChunkedArray const* mColumn;
249  mutable int mFirstIndex;
250  mutable int mCurrentChunk;
251 
252  private:
254  auto getCurrentArray() const
255  {
256  std::shared_ptr<arrow::Array> chunkToUse = mColumn->chunk(mCurrentChunk);
257  if constexpr (std::is_same_v<arrow_array_for_t<T>, arrow::FixedSizeListArray>) {
258  chunkToUse = std::dynamic_pointer_cast<arrow::FixedSizeListArray>(chunkToUse)->values();
259  return std::static_pointer_cast<arrow_array_for_t<value_for_t<T>>>(chunkToUse);
260  } else {
261  return std::static_pointer_cast<arrow_array_for_t<T>>(chunkToUse);
262  }
263  }
264 };
