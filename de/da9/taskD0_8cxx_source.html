56  void process(soa::Filtered<soa::Join<aod::HfCandProng2, aod::HFSelD0Candidate>> const& candidates)
57  {
58  for (auto& candidate : candidates) {
59  if (candidate.isSelD0() >= d_selectionFlagD0) {
60  hmass->Fill(InvMassD0(candidate));
61  }
62  if (candidate.isSelD0bar() >= d_selectionFlagD0bar) {
63  hmass->Fill(InvMassD0bar(candidate));
64  }
65  hptcand->Fill(candidate.pt());
66  hptprong0->Fill(candidate.ptProng0());
67  hptprong1->Fill(candidate.ptProng1());
68  hdeclength->Fill(candidate.decayLength());
69  hdeclengthxy->Fill(candidate.decayLengthXY());
70  hd0->Fill(candidate.impactParameter0());
71  hd0->Fill(candidate.impactParameter1());
72  hd0d0->Fill(candidate.impactParameterProduct());
73  hCTS->Fill(CosThetaStarD0(candidate));
74  hCt->Fill(CtD0(candidate));
75  hEta->Fill(candidate.eta());
76  hselectionstatus->Fill(candidate.isSelD0() + (candidate.isSelD0bar() * 2));
77  }
78  }
79 };
