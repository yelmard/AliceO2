71  // complete timeframe with empty readout windows

81  int n_tof_window = row->size();

116 void CompressedDecodingTask::run(ProcessingContext& pc)
117 {
118  LOG(INFO) << "CompressedDecoding run";
119  mTimer.Start(false);
120 
121  if (pc.inputs().getNofParts(0)) {
122  //RS set the 1st orbit of the TF from the O2 header, relying on rdhHandler is not good (in fact, the RDH might be eliminated in the derived data)
123  const auto* dh = o2::header::get<o2::header::DataHeader*>(pc.inputs().getByPos(0).header);
124  mInitOrbit = dh->firstTForbit;
125  }
126 
127  mDecoder.setFirstIR({0, mInitOrbit});
128 
130  for (auto iit = pc.inputs().begin(), iend = pc.inputs().end(); iit != iend; ++iit) {
131  if (!iit.isValid())
132  continue;
133 
135  for (auto const& ref : iit) {
136 
137  const auto* headerIn = DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
138  auto payloadIn = ref.payload;
139  auto payloadInSize = headerIn->payloadSize;
140 
141  DecoderBase::setDecoderBuffer(payloadIn);
142  DecoderBase::setDecoderBufferSize(payloadInSize);
143  DecoderBase::run();
144  }
145  }
146 
147  if (mNCrateOpenTF == 72 && mNCrateOpenTF == mNCrateCloseTF)
148  mHasToBePosted = true;
149 
150  if (mHasToBePosted) {
151  postData(pc);
152  }
153  mTimer.Stop();
154 }

162 void CompressedDecodingTask::rdhHandler(const o2::header::RAWDataHeader* rdh)
163 {
164 
165  // rdh close
166  const auto& rdhr = *rdh;
167  if (RDHUtils::getStop(rdhr) && RDHUtils::getHeartBeatOrbit(rdhr) == o2::raw::HBFUtils::Instance().getNOrbitsPerTF() - 1 + mInitOrbit) {
168  mNCrateCloseTF++;
169  // printf("New TF close RDH %d\n", int(rdh->feeId));
170  return;
171  }
172 
173  // rdh open
174  if ((RDHUtils::getPageCounter(rdhr) == 0) && (RDHUtils::getTriggerType(rdhr) & o2::trigger::TF)) {
175  mNCrateOpenTF++;
176  mInitOrbit = RDHUtils::getHeartBeatOrbit(rdhr); // RSTODO this may be eliminated once the framework will start to propagated the dh.firstTForbit
177  // printf("New TF open RDH %d\n", int(rdh->feeId));
178  }
179 };

183 {

187  }
