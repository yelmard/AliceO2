137  {
138  // Reset the fValues array
139  VarManager::ResetValues(0, VarManager::kNEventWiseVariables, fValues);
140

146  } else

212  void process(MyEventsSelected::iterator const& event, MyBarrelTracks const& tracks)
213  {
214  VarManager::ResetValues(0, VarManager::kNBarrelTrackVariables, fValues);
215  // fill event information which might be needed in histograms that combine track and event properties
216  VarManager::FillEvent<fgEventFillMap>(event, fValues);
217 
218  uint8_t filterMap = uint8_t(0);
219 
220  for (auto& track : tracks) {
221  filterMap = uint8_t(0);
222  VarManager::FillTrack<fgTrackFillMap>(track, fValues);
223  if (event.isEventSelected())
224  fHistMan->FillHistClass("TrackBarrel_BeforeCuts", fValues);
225 
226  if ((track.flags() & (uint64_t(1) << 2)) && // kITSrefit
227  (track.flags() & (uint64_t(1) << 6)) && // kTPCrefit
228  ((track.itsClusterMap() & (uint8_t(1) << 0)) || (track.itsClusterMap() & (uint8_t(1) << 1)))) //SPD any
229  {
230  int i = 0;
231  for (auto cut = fTrackCuts.begin(); cut != fTrackCuts.end(); ++cut, ++i) {
232  if ((*cut).IsSelected(fValues)) {
233  filterMap |= (uint8_t(1) << i);
234  fHistMan->FillHistClass(Form("TrackBarrel_%s", (*cut).GetName()), fValues);
235  }
236  }
237  }
238  trackSel(filterMap);
239  }
240  }
241 };

391 struct DileptonHadronAnalysis {
392  //
393  // This task combines dilepton candidates with a track and could be used for example
394  // in analyses with the dilepton as one of the decay products of a higher mass resonance (e.g. B0 -> Jpsi + K)
395  // or in dilepton + hadron correlations, etc.
396  // It requires the TableReader task to be in the workflow and produce the dilepton table
397  //
398  OutputObj<HistogramManager> fHistMan{"output"};
399  AnalysisCompositeCut* fHadronCut; // TODO: this cut will be moved into the barrel/muon track selection task
400  //NOTE: no cut has been included for dileptons because that can be controlled via the TableReader task and the partition below
401 
402  // use two values array to avoid mixing up the quantities
403  float* fValuesDilepton;
404  float* fValuesHadron;
405 
406  Filter eventFilter = aod::reducedevent::isEventSelected == 1;
407 
408  Partition<aod::Dileptons> selDileptons = aod::reducedpair::charge == 0 && aod::reducedpair::mass > 2.92f && aod::reducedpair::mass<3.16f && aod::reducedpair::pt> 5.0f;
409

415  fValuesHadron = new float[VarManager::kNVars];

417  fHistMan.setObject(new HistogramManager("analysisHistos", "aa", VarManager::kNVars));
418  fHistMan->SetUseDefaultVariableNames(kTRUE);
419  fHistMan->SetDefaultVarNames(VarManager::fgVariableNames, VarManager::fgVariableUnits);
420 
421  DefineHistograms(fHistMan, "DileptonsSelected;HadronsSelected;DileptonHadronInvMass;DileptonHadronCorrelation"); // define all histograms

431  cut1->AddCut(VarManager::kPt, 4.0, 20.0);
432  cut1->AddCut(VarManager::kEta, -0.9, 0.9);

442  void process(soa::Filtered<MyEventsVtxCovSelected>::iterator const& event, MyBarrelTracks const& hadrons, aod::Dileptons const& dileptons)
443  {
444  VarManager::ResetValues(0, VarManager::kNVars, fValuesHadron);
445  VarManager::ResetValues(0, VarManager::kNVars, fValuesDilepton);
446  // fill event information which might be needed in histograms that combine track/pair and event properties
447  VarManager::FillEvent<fgEventFillMap>(event, fValuesHadron);
448  VarManager::FillEvent<fgEventFillMap>(event, fValuesDilepton); // TODO: check if needed (just for dilepton QA which might be depending on event wise variables)
449 
450  // loop once over dileptons for QA purposes
451  for (auto dilepton : selDileptons) {
452  VarManager::FillTrack<fgDileptonFillMap>(dilepton, fValuesDilepton);
453  fHistMan->FillHistClass("DileptonsSelected", fValuesDilepton);
454  }
455 
456  // loop over hadrons
457  for (auto& hadron : hadrons) {
458  VarManager::FillTrack<fgTrackFillMap>(hadron, fValuesHadron);
459  if (!fHadronCut->IsSelected(fValuesHadron)) // TODO: this will be moved to a partition when the selection will be done in the barrel/muon track selection
460  continue;
461 
462  fHistMan->FillHistClass("HadronsSelected", fValuesHadron);
463 
464  for (auto dilepton : selDileptons) {
465  // TODO: At the moment there is no check on whether this hadron is one of the dilepton daughters!
466  VarManager::FillDileptonHadron(dilepton, hadron, fValuesHadron);
467  fHistMan->FillHistClass("DileptonHadronInvMass", fValuesHadron);
468  fHistMan->FillHistClass("DileptonHadronCorrelation", fValuesHadron);
469  }
470  }
471  }
