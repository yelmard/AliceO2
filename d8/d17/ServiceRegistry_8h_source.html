151  // Lookup a given @a typeHash for a given @a threadId at
152  // a unique (per typeHash) location. There might
153  // be other typeHash which sit in the same place, but
154  // the if statement will rule them out. As long as
155  // only one thread writes in a given i + id location
156  // as guaranteed by the atomic, mServicesKey[i + id] will
157  // either be 0 or the final value.
158  // This method should NEVER register a new service, event when requested.
159  int getPos(uint32_t typeHash, uint64_t threadId) const
160  {
161  auto id = typeHash & MAX_SERVICES_MASK;
162  auto threadHashId = (typeHash ^ threadId) & MAX_SERVICES_MASK;
163  std::atomic_thread_fence(std::memory_order_acquire);
164  for (uint8_t i = 0; i < MAX_DISTANCE; ++i) {
165  if (mServicesKey[i + threadHashId] == typeHash) {
166  return i + threadHashId;
167  }
168  }
169  return -1;
170  }
171 
172  // Basic, untemplated API. This will require explicitly
173  // providing the @a typeHash for the Service type,
174  // there @a threadId of the thread asking the service
175  // and the @a kind of service. This
176  // method might trigger the registering of a new service
177  // if the service is not a stream service and the global
178  // zero service is available.
179  // Use this API only if you know what you are doing.
180  void* get(uint32_t typeHash, uint64_t threadId, ServiceKind kind, char const* name = nullptr) const
181  {
182  // Look for the service. If found, return it.
183  // Notice how due to threading issues, we might
184  // find it with getPos, but the value can still
185  // be nullptr.
186  auto pos = getPos(typeHash, threadId);
187  if (pos != -1) {
188  void* ptr = mServicesValue[pos];
189  if (ptr) {
190  return ptr;
191  }
192  }
193  // We are looking up a service which is not of
194  // stream kind and was not looked up by this thread
195  // before.
196  if (threadId != 0) {
197  int pos = getPos(typeHash, 0);
198  if (pos != -1 && kind != ServiceKind::Stream) {
199  registerService(typeHash, mServicesValue[pos], kind, threadId, name);
200  }
201  return mServicesValue[pos];
202  }
203  // If we are here it means we never registered a
204  // service for the 0 thread (i.e. the main thread).
205  return nullptr;
206  }

223  template <class I, class C, enum ServiceKind K = ServiceKind::Serial>
224  void registerService(C* service)
225  {
226  // This only works for concrete implementations of the type T.
227  // We need type elision as we do not want to know all the services in
228  // advance
229  static_assert(std::is_base_of<I, C>::value == true,
230  "Registered service is not derived from declared interface");
231  constexpr hash_type typeHash = TypeIdHelpers::uniqueId<I>();
232  auto tid = std::this_thread::get_id();
233  std::hash<std::thread::id> hasher;
234  ServiceRegistry::registerService(typeHash, reinterpret_cast<void*>(service), K, hasher(tid), typeid(C).name());
235  }

254  template <typename T>
255  std::enable_if_t<std::is_const_v<T> == false, bool> active() const
256  {
257  constexpr auto typeHash = TypeIdHelpers::uniqueId<T>();
258  auto tid = std::this_thread::get_id();
259  std::hash<std::thread::id> hasher;
260  if (this->getPos(typeHash, 0) != -1) {
261  return true;
262  }
263  auto result = this->getPos(typeHash, hasher(tid)) != -1;
264  return result;
265  }

270  template <typename T>
271  T& get() const
272  {
273  constexpr auto typeHash = TypeIdHelpers::uniqueId<T>();
274  auto tid = std::this_thread::get_id();
275  std::hash<std::thread::id> hasher;
276  auto ptr = this->get(typeHash, hasher(tid), ServiceKind::Serial, typeid(T).name());
277  if (O2_BUILTIN_LIKELY(ptr != nullptr)) {
278  if constexpr (std::is_const_v<T>) {
279  return *reinterpret_cast<T const*>(ptr);
280  } else {
281  return *reinterpret_cast<T*>(ptr);
282  }
283  }
284  throw std::runtime_error(std::string("Unable to find service of kind ") +
285  typeid(T).name() +
286  ". Make sure you use const / non-const correctly.");
287  }
