34  Configurable<double> d_bz{"d_bz", 5.0, "bz field"};

40  void process(aod::Collision const& collision,
41  soa::Join<aod::Tracks, aod::TracksCov, aod::TracksExtra> const& tracks)
42  {
43  Point3D<float> vtxXYZ(collision.posX(), collision.posY(), collision.posZ());
44  for (auto& track : tracks) {
45  int status = 1; // selection flag
46  if (b_dovalplots)
47  hpt_nocuts->Fill(track.pt());
48  if (track.pt() < ptmintrack)
49  status = 0;
50  UChar_t clustermap_0 = track.itsClusterMap();
51  bool isselected_0 = track.tpcNClsFound() >= d_tpcnclsfound && track.flags() & 0x4;
52  isselected_0 = isselected_0 && (TESTBIT(clustermap_0, 0) || TESTBIT(clustermap_0, 1));
53  if (!isselected_0)
54  status = 0;
55  array<float, 2> dca;
56  auto trackparvar0 = getTrackParCov(track);
57  bool isprop = trackparvar0.propagateParamToDCA(vtxXYZ, d_bz, &dca, 100.); // get impact parameters
58  if (!isprop)
59  status = 0;
60  if (abs(dca[0]) < dcatoprimxymin)
61  status = 0;
62  if (b_dovalplots) {
63  if (status == 1) {
64  hpt_cuts->Fill(track.pt());
65  hdcatoprimxy_cuts->Fill(dca[0]);
66  }
67  }
68  rowSelectedTrack(status, dca[0], dca[1]);
69  }
70  }
71 };
