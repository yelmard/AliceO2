42 template <class DataBlockType>
43 class DataBlockReader
44 {
45  public:
46  DataBlockReader() = default;
47  ~DataBlockReader() = default;
48  typedef DataBlockType DataBlock;
49  size_t decodeBlocks(const gsl::span<const uint8_t> binaryPayload, std::vector<DataBlockType>& vecDataBlocks)
50  {
51  size_t srcPos = 0;
52  while (srcPos < binaryPayload.size()) { //checking element
53  DataBlockType dataBlock;
54  dataBlock.decodeBlock(binaryPayload, srcPos);
55  srcPos += dataBlock.mSize;
56  if (dataBlock.isCorrect())
57  vecDataBlocks.push_back(dataBlock); //change to in-place construction? TODO
58  }
59  return srcPos;
60  }
61 };

73  std::map<InteractionRecord, DigitBlockType> mMapDigits;
74  template <class DataBlockType>
75  void pushDataBlock(gsl::span<const uint8_t> payload, int linkID)
76  {
77  std::vector<DataBlockType> vecDataBlocks;
78  auto srcPos = DataBlockReader<DataBlockType>::decodeBlocks(payload, vecDataBlocks);
79  for (auto& dataBlock : vecDataBlocks) {
80  auto intRec = dataBlock.getInteractionRecord();
81  auto [digitIter, isNew] = mMapDigits.try_emplace(intRec, intRec);
82  digitIter->second.template proccess<DataBlockType>(dataBlock, linkID);
83  }
84  }
85 };
