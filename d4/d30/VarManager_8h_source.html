217 template <uint32_t fillMap, typename T>
218 void VarManager::FillEvent(T const& event, float* values)
219 {
220  if (!values)
221  values = fgValues;
222 
223  if constexpr ((fillMap & BC) > 0) {
224  values[kRunNo] = event.bc().runNumber(); // accessed via Collisions table
225  values[kBC] = event.bc().globalBC();
226  }
227 
228  if constexpr ((fillMap & Collision) > 0) {
229  values[kVtxX] = event.posX();
230  values[kVtxY] = event.posY();
231  values[kVtxZ] = event.posZ();
232  values[kVtxNcontrib] = event.numContrib();
233  values[kCollisionTime] = event.collisionTime();
234  values[kVtxCovXX] = event.covXX();
235  values[kVtxCovXY] = event.covXY();
236  values[kVtxCovXZ] = event.covXZ();
237  values[kVtxCovYY] = event.covYY();
238  values[kVtxCovYZ] = event.covYZ();
239  values[kVtxCovZZ] = event.covZZ();
240  values[kVtxChi2] = event.chi2();
241  }
242 
243  // TODO: need to add EvSels and Cents tables, etc. in case of the central data model
244 
245  if constexpr ((fillMap & ReducedEvent) > 0) {
246  values[kRunNo] = event.runNumber();
247  values[kVtxX] = event.posX();
248  values[kVtxY] = event.posY();
249  values[kVtxZ] = event.posZ();
250  values[kVtxNcontrib] = event.numContrib();
251  }
252 
253  if constexpr ((fillMap & ReducedEventExtended) > 0) {
254  values[kBC] = event.globalBC();
255  values[kCentVZERO] = event.centV0M();
256  }
257 
258  if constexpr ((fillMap & ReducedEventVtxCov) > 0) {
259  values[kVtxCovXX] = event.covXX();
260  values[kVtxCovXY] = event.covXY();
261  values[kVtxCovXZ] = event.covXZ();
262  values[kVtxCovYY] = event.covYY();
263  values[kVtxCovYZ] = event.covYZ();
264  values[kVtxCovZZ] = event.covZZ();
265  values[kVtxChi2] = event.chi2();
266  }
267 
268  FillEventDerived(values);
269 }

271 template <uint32_t fillMap, typename T>
272 void VarManager::FillTrack(T const& track, float* values)
273 {
274  if (!values)
275  values = fgValues;
276 
277  if constexpr ((fillMap & Track) > 0) {
278  values[kPt] = track.pt();
279  values[kEta] = track.eta();
280  values[kPhi] = track.phi();
281  values[kCharge] = track.charge();
282  }
283 
284  if constexpr ((fillMap & TrackExtra) > 0) {
285  values[kPin] = track.tpcInnerParam();
286  if (fgUsedVars[kITSncls])
287  values[kITSncls] = track.itsNCls(); // dynamic column
288  values[kITSchi2] = track.itsChi2NCl();
289  values[kTPCncls] = track.tpcNClsFound();
290  values[kTPCchi2] = track.tpcChi2NCl();
291  values[kTrackLength] = track.length();
292  values[kTrackDCAxy] = track.dcaXY();
293  values[kTrackDCAz] = track.dcaZ();
294  }
295 
296  if constexpr ((fillMap & TrackCov) > 0) {
297  values[kTrackCYY] = track.cYY();
298  values[kTrackCZZ] = track.cZZ();
299  values[kTrackCSnpSnp] = track.cSnpSnp();
300  values[kTrackCTglTgl] = track.cTglTgl();
301  values[kTrackC1Pt21Pt2] = track.c1Pt21Pt2();
302  }
303 
304  if constexpr ((fillMap & ReducedTrack) > 0) {
305  values[kPt] = track.pt();
306  if (fgUsedVars[kPx])
307  values[kPx] = track.px();
308  if (fgUsedVars[kPy])
309  values[kPy] = track.py();
310  if (fgUsedVars[kPz])
311  values[kPz] = track.pz();
312  values[kEta] = track.eta();
313  values[kPhi] = track.phi();
314  values[kCharge] = track.charge();
315  }
316 
317  if constexpr ((fillMap & ReducedTrackBarrel) > 0) {
318  values[kPin] = track.tpcInnerParam();
319  if (fgUsedVars[kITSncls]) { // TODO: add the central data model dynamic column to the reduced table
320  values[kITSncls] = 0.0;
321  for (int i = 0; i < 6; ++i)
322  values[kITSncls] += ((track.itsClusterMap() & (1 << i)) ? 1 : 0);
323  }
324  values[kITSchi2] = track.itsChi2NCl();
325  values[kTPCncls] = track.tpcNClsFound();
326  values[kTPCchi2] = track.tpcChi2NCl();
327  values[kTrackLength] = track.length();
328  values[kTrackDCAxy] = track.dcaXY();
329  values[kTrackDCAz] = track.dcaZ();
330  }
331 
332  if constexpr ((fillMap & ReducedTrackBarrelCov) > 0) {
333  values[kTrackCYY] = track.cYY();
334  values[kTrackCZZ] = track.cZZ();
335  values[kTrackCSnpSnp] = track.cSnpSnp();
336  values[kTrackCTglTgl] = track.cTglTgl();
337  values[kTrackC1Pt21Pt2] = track.c1Pt21Pt2();
338  }
339 
340  if constexpr ((fillMap & ReducedTrackBarrelPID) > 0) {
341  values[kTPCnSigmaEl] = track.tpcNSigmaEl();
342  values[kTPCnSigmaMu] = track.tpcNSigmaMu();
343  values[kTPCnSigmaPi] = track.tpcNSigmaPi();
344  values[kTPCnSigmaKa] = track.tpcNSigmaKa();
345  values[kTPCnSigmaPr] = track.tpcNSigmaPr();
346  values[kTOFnSigmaEl] = track.tofNSigmaEl();
347  values[kTOFnSigmaMu] = track.tofNSigmaMu();
348  values[kTOFnSigmaPi] = track.tofNSigmaPi();
349  values[kTOFnSigmaKa] = track.tofNSigmaKa();
350  values[kTOFnSigmaPr] = track.tofNSigmaPr();
351  values[kTPCsignal] = track.tpcSignal();
352  values[kTRDsignal] = track.trdSignal();
353  values[kTOFsignal] = track.tofSignal();
354  values[kTOFbeta] = track.beta();
355  }
356 
357  if constexpr ((fillMap & ReducedTrackMuon) > 0) {
358  values[kMuonInvBendingMomentum] = track.inverseBendingMomentum();
359  values[kMuonThetaX] = track.thetaX();
360  values[kMuonThetaY] = track.thetaY();
361  values[kMuonZMu] = track.zMu();
362  values[kMuonBendingCoor] = track.bendingCoor();
363  values[kMuonNonBendingCoor] = track.nonBendingCoor();
364  values[kMuonRAtAbsorberEnd] = track.rAtAbsorberEnd();
365  values[kMuonChi2] = track.chi2();
366  values[kMuonChi2MatchTrigger] = track.chi2MatchTrigger();
367  }
368 
369  if constexpr ((fillMap & Pair) > 0) {
370  values[kMass] = track.mass();
371  }
372 
373  FillTrackDerived(values);
374 }

376 template <typename T>
377 void VarManager::FillPair(T const& t1, T const& t2, float* values)
378 {
379  if (!values)
380  values = fgValues;
381 
382  float mass1 = fgkElectronMass;
383  float mass2 = fgkElectronMass;
384 
385  bool isMuon1 = t1.filteringFlags() & (1 << 0);
386  bool isMuon2 = t2.filteringFlags() & (1 << 0);
387 
388  if (isMuon1)
389  mass1 = fgkMuonMass;
390  else
391  mass1 = fgkElectronMass;
392 
393  if (isMuon2)
394  mass2 = fgkMuonMass;
395  else
396  mass2 = fgkElectronMass;
397 
398  ROOT::Math::PtEtaPhiMVector v1(t1.pt(), t1.eta(), t1.phi(), mass1);
399  ROOT::Math::PtEtaPhiMVector v2(t2.pt(), t2.eta(), t2.phi(), mass2);
400  ROOT::Math::PtEtaPhiMVector v12 = v1 + v2;
401  values[kMass] = v12.M();
402  values[kPt] = v12.Pt();
403  values[kEta] = v12.Eta();
404  values[kPhi] = v12.Phi();
405  values[kRap] = -v12.Rapidity();
406 }

408 template <typename T1, typename T2>
409 void VarManager::FillDileptonHadron(T1 const& dilepton, T2 const& hadron, float* values, float hadronMass)
410 {
411  if (!values)
412  values = fgValues;
413 
414  if (fgUsedVars[kPairMass] || fgUsedVars[kPairPt] || fgUsedVars[kPairEta] || fgUsedVars[kPairPhi]) {
415  ROOT::Math::PtEtaPhiMVector v1(dilepton.pt(), dilepton.eta(), dilepton.phi(), dilepton.mass());
416  ROOT::Math::PtEtaPhiMVector v2(hadron.pt(), hadron.eta(), hadron.phi(), hadronMass);
417  ROOT::Math::PtEtaPhiMVector v12 = v1 + v2;
418  values[kPairMass] = v12.M();
419  values[kPairPt] = v12.Pt();
420  values[kPairEta] = v12.Eta();
421  values[kPairPhi] = v12.Phi();
422  }
423  if (fgUsedVars[kDeltaPhi]) {
424  double delta = dilepton.phi() - hadron.phi();
425  if (delta > 3.0 / 2.0 * TMath::Pi())
426  delta -= 2.0 * TMath::Pi();
427  if (delta < -0.5 * TMath::Pi())
428  delta += 2.0 * TMath::Pi();
429  values[kDeltaPhi] = delta;
430  }
431  if (fgUsedVars[kDeltaPhiSym]) {
432  double delta = TMath::Abs(dilepton.phi() - hadron.phi());
433  if (delta > TMath::Pi())
434  delta = 2 * TMath::Pi() - delta;
435  values[kDeltaPhiSym] = delta;
436  }
437  if (fgUsedVars[kDeltaEta])
438  values[kDeltaEta] = dilepton.eta() - hadron.eta();
439 }
