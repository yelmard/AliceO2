66  }

70 struct lambdakzeroconsumer {
71  OutputObj<TH3F> h3dMassK0Short{TH3F("h3dMassK0Short", "", 20, 0, 100, 200, 0, 10, 200, 0.450, 0.550)};
72  OutputObj<TH3F> h3dMassLambda{TH3F("h3dMassLambda", "", 20, 0, 100, 200, 0, 10, 200, 1.115 - 0.100, 1.115 + 0.100)};
73  OutputObj<TH3F> h3dMassAntiLambda{TH3F("h3dMassAntiLambda", "", 20, 0, 100, 200, 0, 10, 200, 1.115 - 0.100, 1.115 + 0.100)};
74 
75  //Selection criteria
76  Configurable<double> v0cospa{"v0cospa", 0.995, "V0 CosPA"}; //double -> N.B. dcos(x)/dx = 0 at x=0)
77  Configurable<float> dcav0dau{"dcav0dau", 1.0, "DCA V0 Daughters"};
78  Configurable<float> dcanegtopv{"dcanegtopv", .1, "DCA Neg To PV"};
79  Configurable<float> dcapostopv{"dcapostopv", .1, "DCA Pos To PV"};
80  Configurable<float> v0radius{"v0radius", 5.0, "v0radius"};
81 
82  Filter preFilterV0 = aod::v0data::dcapostopv > dcapostopv&&
83  aod::v0data::dcanegtopv > dcanegtopv&& aod::v0data::dcaV0daughters < dcav0dau;
84 
85  void process(soa::Join<aod::Collisions, aod::EvSels, aod::Cents>::iterator const& collision, soa::Filtered<soa::Join<aod::V0s, aod::V0DataExt>> const& fullV0s)
86  {
87  if (!collision.alias()[kINT7])
88  return;
89  if (!collision.sel7())
90  return;
91 
92  for (auto& v0 : fullV0s) {
93  //FIXME: could not find out how to filter cosPA and radius variables (dynamic columns)
94  if (v0.v0radius() > v0radius && v0.v0cosPA(collision.posX(), collision.posY(), collision.posZ()) > v0cospa) {
95  if (TMath::Abs(v0.yLambda()) < 0.5) {
96  h3dMassLambda->Fill(collision.centV0M(), v0.pt(), v0.mLambda());
97  h3dMassAntiLambda->Fill(collision.centV0M(), v0.pt(), v0.mAntiLambda());
98  }
99  if (TMath::Abs(v0.yK0Short()) < 0.5)
100  h3dMassK0Short->Fill(collision.centV0M(), v0.pt(), v0.mK0Short());
101  }
102  }
103  }
104 };
