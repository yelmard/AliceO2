145 inline void PVertexer::applyConstraint(VertexSeed& vtxSeed) const
146 {
147  // impose meanVertex constraint, i.e. account terms
148  // (V_i-Constrain_i)^2/sig2constr_i for i=X,Y in the fit chi2 definition
149  vtxSeed.cxx += mXYConstraintInvErr[0];
150  vtxSeed.cyy += mXYConstraintInvErr[1];
151  vtxSeed.cx0 += mXYConstraintInvErr[0] * mMeanVertex.getX();
152  vtxSeed.cy0 += mXYConstraintInvErr[1] * mMeanVertex.getY();
153 }

155 //___________________________________________________________________
156 inline bool PVertexer::upscaleSigma(VertexSeed& vtxSeed) const
157 {
158  // scale upward the scaleSigma2 if needes
159  if (vtxSeed.scaleSigma2 < mPVParams->maxScale2) {
160  auto s = vtxSeed.scaleSigma2 * mPVParams->upscaleFactor;
161  vtxSeed.setScale(s > mPVParams->maxScale2 ? mPVParams->maxScale2 : s, mTukey2I);
162  return true;
163  }
164  return false;
165 }
