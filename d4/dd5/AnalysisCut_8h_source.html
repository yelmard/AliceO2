74 void AnalysisCut::AddCut(int var, T1 cutLow, T2 cutHigh, bool exclude,
75  int dependentVar, float depCutLow, float depCutHigh, bool depCutExclude,
76  int dependentVar2, float depCut2Low, float depCut2High, bool depCut2Exclude)
77 {
78  //
79  // Add a cut
80  //
81  CutContainer cut = {};
82 
83  if constexpr (std::is_same_v<T1, TF1*>) {
84  if (dependentVar < 0)
85  return;
86  cut.fFuncLow = cutLow;
87  cut.fLow = -9999.0;
88  } else {
89  cut.fFuncLow = nullptr;
90  cut.fLow = cutLow;
91  }
92  if constexpr (std::is_same_v<T2, TF1*>) {
93  if (dependentVar < 0)
94  return;
95  cut.fFuncHigh = cutHigh;
96  cut.fHigh = -9999.0;
97  } else {
98  cut.fFuncHigh = nullptr;
99  cut.fHigh = cutHigh;
100  }
101  cut.fVar = var;
102  cut.fExclude = exclude;
103  fgUsedVars.push_back(var);
104 
105  cut.fDepVar = dependentVar;
106  cut.fDepLow = depCutLow;
107  cut.fDepHigh = depCutHigh;
108  cut.fDepExclude = depCutExclude;
109  if (dependentVar != -1)
110  fgUsedVars.push_back(dependentVar);
111 
112  cut.fDepVar2 = dependentVar2;
113  cut.fDep2Low = depCut2Low;
114  cut.fDep2High = depCut2High;
115  cut.fDep2Exclude = depCut2Exclude;
116  if (dependentVar2 != -1)
117  fgUsedVars.push_back(dependentVar2);
118 
119  fCuts.push_back(cut);
120 }
