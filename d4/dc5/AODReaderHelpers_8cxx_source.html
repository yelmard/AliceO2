44 AlgorithmSpec AODReaderHelpers::aodSpawnerCallback(std::vector<InputSpec> requested)
45 {
46  return AlgorithmSpec::InitCallback{[requested](InitContext& ic) {
47  auto& callbacks = ic.services().get<CallbackService>();
48  auto endofdatacb = [](EndOfStreamContext& eosc) {
49  auto& control = eosc.services().get<ControlService>();
50  control.endOfStream();
51  control.readyToQuit(QuitRequest::Me);
52  };
53  callbacks.set(CallbackService::Id::EndOfStream, endofdatacb);
54 
55  return [requested](ProcessingContext& pc) {
56  auto outputs = pc.outputs();
57  // spawn tables
58  for (auto& input : requested) {
59  auto description = std::visit(
60  overloaded{
61  [](ConcreteDataMatcher const& matcher) { return matcher.description; },
62  [](auto&&) { return header::DataDescription{""}; }},
63  input.matcher);
64 
65  auto origin = std::visit(
66  overloaded{
67  [](ConcreteDataMatcher const& matcher) { return matcher.origin; },
68  [](auto&&) { return header::DataOrigin{""}; }},
69  input.matcher);
70 
71  auto maker = [&](auto metadata) {
72  using metadata_t = decltype(metadata);
73  using expressions = typename metadata_t::expression_pack_t;
74  auto original_table = pc.inputs().get<TableConsumer>(input.binding)->asArrowTable();
75  return o2::soa::spawner(expressions{}, original_table.get());
76  };
77 
78  if (description == header::DataDescription{"TRACK:PAR"}) {
79  outputs.adopt(Output{origin, description}, maker(o2::aod::TracksExtensionMetadata{}));
80  } else if (description == header::DataDescription{"TRACK:PARCOV"}) {
81  outputs.adopt(Output{origin, description}, maker(o2::aod::TracksCovExtensionMetadata{}));
82  } else if (description == header::DataDescription{"MUON"}) {
83  outputs.adopt(Output{origin, description}, maker(o2::aod::MuonsExtensionMetadata{}));
84  } else {
85  throw std::runtime_error("Not an extended table");
86  }
87  }
88  };
89  }};
90 }

92 AlgorithmSpec AODReaderHelpers::rootFileReaderCallback()
93 {
94  auto callback = AlgorithmSpec{adaptStateful([](ConfigParamRegistry const& options,
95  DeviceSpec const& spec) {
96  auto filename = options.get<std::string>("aod-file");
97 
98  // create a DataInputDirector
99  auto didir = std::make_shared<DataInputDirector>(filename);
100  if (options.isSet("json-file")) {
101  auto jsonFile = options.get<std::string>("json-file");
102  if (!didir->readJson(jsonFile)) {
103  LOGP(ERROR, "Check the JSON document! Can not be properly parsed!");
104  }
105  }
106 
107  // get the run time watchdog
108  auto* watchdog = new RuntimeWatchdog(options.get<int64_t>("time-limit"));
109 
110  // create list of requested tables
111  std::vector<OutputRoute> requestedTables(spec.outputs);
112 
113  auto counter = std::make_shared<int>(0);
114  return adaptStateless([requestedTables,
115  watchdog,
116  didir](DataAllocator& outputs, ControlService& control, DeviceSpec const& device) {
117  // check if RuntimeLimit is reached
118  if (!watchdog->update()) {
119  LOGP(INFO, "Run time exceeds run time limit of {} seconds!", watchdog->runTimeLimit);
120  LOGP(INFO, "Stopping after time frame {}.", watchdog->numberTimeFrames - 1);
121  didir->closeInputFiles();
122  control.endOfStream();
123  control.readyToQuit(QuitRequest::All);
124  return;
125  }
126 
127  // Each parallel reader reads the files whose index is associated to
128  // their inputTimesliceId
129  assert(device.inputTimesliceId < device.maxInputTimeslices);
130  size_t fi = (watchdog->numberTimeFrames * device.maxInputTimeslices) + device.inputTimesliceId;
131 
132  // check if EoF is reached
133  if (didir->atEnd(fi)) {
134  LOGP(INFO, "All input files processed");
135  didir->closeInputFiles();
136  control.endOfStream();
137  control.readyToQuit(QuitRequest::Me);
138  return;
139  }
140 
141  // loop over requested tables
142  for (auto route : requestedTables) {
143 
144  // create a TreeToTable object
145  auto concrete = DataSpecUtils::asConcreteDataMatcher(route.matcher);
146  auto dh = header::DataHeader(concrete.description, concrete.origin, concrete.subSpec);
147 
148  auto tr = didir->getDataTree(dh, fi);
149  if (!tr) {
150  char* table;
151  sprintf(table, "%s/%s/%" PRIu32, concrete.origin.str, concrete.description.str, concrete.subSpec);
152  LOGP(ERROR, "Error while retrieving the tree for \"{}\"!", table);
153  return;
154  }
155 
156  auto o = Output(dh);
157  auto& t2t = outputs.make<TreeToTable>(o, tr);
158 
159  // add branches to read
160  auto colnames = aod::datamodel::getColumnNames(dh);
161  if (colnames.size() == 0) {
162  t2t.addAllColumns();
163  } else {
164  for (auto colname : colnames) {
165  t2t.addColumn(colname.c_str());
166  }
167  }
168 
169  // fill the table
170  t2t.fill();
171  }
172  });
173  })};
174 
175  return callback;
176 }
