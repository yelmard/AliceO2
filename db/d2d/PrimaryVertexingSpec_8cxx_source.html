43 void PrimaryVertexingSpec::run(ProcessingContext& pc)
44 {
45  double timeCPU0 = mTimer.CpuTime(), timeReal0 = mTimer.RealTime();
46  mTimer.Start(false);
47  const auto tracksITSTPC = pc.inputs().get<gsl::span<o2::dataformats::TrackTPCITS>>("match");
48  gsl::span<const o2::MCCompLabel> lblITS, lblTPC;
49  gsl::span<const o2::ft0::RecPoints> ft0Data;
50  if (mValidateWithFT0) {
51  ft0Data = pc.inputs().get<gsl::span<o2::ft0::RecPoints>>("fitInfo");
52  }
53  if (mUseMC) {
54  lblITS = pc.inputs().get<gsl::span<o2::MCCompLabel>>("lblITS");
55  lblTPC = pc.inputs().get<gsl::span<o2::MCCompLabel>>("lblTPC");
56  }
57  std::vector<PVertex> vertices;
58  std::vector<GIndex> vertexTrackIDs;
59  std::vector<V2TRef> v2tRefs;
60  std::vector<o2::MCEventLabel> lblVtx;
61 
62  // RS FIXME this will not have effect until the 1st orbit is propagated, until that will work only for TF starting at orbit 0
63  const auto* dh = o2::header::get<o2::header::DataHeader*>(pc.inputs().get("match").header);
64  mVertexer.setStartIR({0, dh->firstTForbit});
65 
66  mVertexer.process(tracksITSTPC, ft0Data, vertices, vertexTrackIDs, v2tRefs, lblITS, lblTPC, lblVtx);
67  pc.outputs().snapshot(Output{"GLO", "PVTX", 0, Lifetime::Timeframe}, vertices);
68  pc.outputs().snapshot(Output{"GLO", "PVTX_CONTIDREFS", 0, Lifetime::Timeframe}, v2tRefs);
69  pc.outputs().snapshot(Output{"GLO", "PVTX_CONTID", 0, Lifetime::Timeframe}, vertexTrackIDs);
70 
71  if (mUseMC) {
72  pc.outputs().snapshot(Output{"GLO", "PVTX_MCTR", 0, Lifetime::Timeframe}, lblVtx);
73  }
74 
75  mTimer.Stop();
76  LOG(INFO) << "Found " << vertices.size() << " primary vertices, timing: CPU: "
77  << mTimer.CpuTime() - timeCPU0 << " Real: " << mTimer.RealTime() - timeReal0 << " s";
78 }

86 DataProcessorSpec getPrimaryVertexingSpec(bool validateWithFT0, bool useMC)
87 {
88  std::vector<InputSpec> inputs;
89  std::vector<OutputSpec> outputs;
90 
91  inputs.emplace_back("match", "GLO", "TPCITS", 0, Lifetime::Timeframe);
92  if (validateWithFT0) {
93  inputs.emplace_back("fitInfo", "FT0", "RECPOINTS", 0, Lifetime::Timeframe);
94  }
95 
96  outputs.emplace_back("GLO", "PVTX", 0, Lifetime::Timeframe);
97  outputs.emplace_back("GLO", "PVTX_CONTID", 0, Lifetime::Timeframe);
98  outputs.emplace_back("GLO", "PVTX_CONTIDREFS", 0, Lifetime::Timeframe);
99 
100  if (useMC) {
101  inputs.emplace_back("lblITS", "GLO", "TPCITS_ITSMC", 0, Lifetime::Timeframe);
102  inputs.emplace_back("lblTPC", "GLO", "TPCITS_TPCMC", 0, Lifetime::Timeframe);
103  outputs.emplace_back("GLO", "PVTX_MCTR", 0, Lifetime::Timeframe);
104  }
105 
106  return DataProcessorSpec{
107  "primary-vertexing",
108  inputs,
109  outputs,
110  AlgorithmSpec{adaptFromTask<PrimaryVertexingSpec>(validateWithFT0, useMC)},
111  Options{}};
112 }
