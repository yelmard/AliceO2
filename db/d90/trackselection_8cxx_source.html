29 // Default track selection requiring one hit in the SPD
30 TrackSelection getGlobalTrackSelection()
31 {
32  TrackSelection selectedTracks;
33  selectedTracks.SetPtRange(0.1f, 1e10f);
34  selectedTracks.SetEtaRange(-0.8f, 0.8f);
35  selectedTracks.SetRequireITSRefit(true);
36  selectedTracks.SetRequireTPCRefit(true);
37  selectedTracks.SetMinNCrossedRowsTPC(70);
38  selectedTracks.SetMinNCrossedRowsOverFindableClustersTPC(0.8f);
39  selectedTracks.SetMaxChi2PerClusterTPC(4.f);
40  selectedTracks.SetRequireHitsInITSLayers(1, {0, 1}); // one hit in any SPD layer
41  selectedTracks.SetMaxChi2PerClusterITS(36.f);
42  selectedTracks.SetMaxDcaXYPtDep([](float pt) { return 0.0105f + 0.0350f / pow(pt, 1.1f); });
43  selectedTracks.SetMaxDcaZ(2.f);
44  return selectedTracks;
45 }

47 // Default track selection requiring no hit in the SPD and one in the innermost
48 // SDD -> complementary tracks to global selection
49 TrackSelection getGlobalTrackSelectionSDD()
50 {
51  TrackSelection selectedTracks = getGlobalTrackSelection();
52  selectedTracks.ResetITSRequirements();
53  selectedTracks.SetRequireNoHitsInITSLayers({0, 1}); // no hit in SPD layers
54  selectedTracks.SetRequireHitsInITSLayers(1, {2}); // one hit in first SDD layer
55  return selectedTracks;
56 }

58 // Default track selection requiring a cluster matched in TOF
59 TrackSelection getGlobalTrackSelectionwTOF()
60 {
61  TrackSelection selectedTracks = getGlobalTrackSelection();
62  selectedTracks.SetRequireTOF(true);
63  return selectedTracks;
64 }

71 struct TrackExtensionTask {
72 
73  Produces<aod::TracksExtended> extendedTrackQuantities;
74 
75  void process(aod::Collision const& collision, aod::FullTracks const& tracks)
76  {
77  float sinAlpha = 0.f;
78  float cosAlpha = 0.f;
79  float globalX = 0.f;
80  float globalY = 0.f;
81  float dcaXY = 0.f;
82  float dcaZ = 0.f;
83 
84  for (auto& track : tracks) {
85 
86  sinAlpha = sin(track.alpha());
87  cosAlpha = cos(track.alpha());
88  globalX = track.x() * cosAlpha - track.y() * sinAlpha;
89  globalY = track.x() * sinAlpha + track.y() * cosAlpha;
90 
91  dcaXY = track.charge() * sqrt(pow((globalX - collision.posX()), 2) +
92  pow((globalY - collision.posY()), 2));
93  dcaZ = track.charge() * sqrt(pow(track.z() - collision.posZ(), 2));
94 
95  extendedTrackQuantities(dcaXY, dcaZ);
96  }
97  }
98 };

105 struct TrackSelectionTask {
106  Produces<aod::TrackSelection> filterTable;
107 
108  TrackSelection globalTracks;
109  TrackSelection globalTracksSDD;
110  TrackSelection globalTrackswTOF;
111 
112  void init(InitContext&)
113  {
114  globalTracks = getGlobalTrackSelection();
115  globalTracksSDD = getGlobalTrackSelectionSDD();
116  globalTrackswTOF = getGlobalTrackSelectionwTOF();
117  }
118 
119  void process(soa::Join<aod::FullTracks, aod::TracksExtended> const& tracks)
120  {
121  for (auto& track : tracks) {
122  filterTable((uint8_t)globalTracks.IsSelected(track),
123  (uint8_t)globalTracksSDD.IsSelected(track),
124  (uint8_t)globalTrackswTOF.IsSelected(track));
125  }
126  }
127 };
