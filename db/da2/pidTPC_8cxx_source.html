26 struct pidTPCTask {
27  Produces<aod::pidRespTPC> tpcpid;
28  tpc::Response resp = tpc::Response();
29  Service<o2::ccdb::BasicCCDBManager> ccdb;
30  Configurable<std::string> paramfile{"param-file", "", "Path to the parametrization object, if emtpy the parametrization is not taken from file"};
31  Configurable<std::string> signalname{"param-signal", "BetheBloch", "Name of the parametrization for the expected signal, used in both file and CCDB mode"};
32  Configurable<std::string> sigmaname{"param-sigma", "TPCReso", "Name of the parametrization for the expected sigma, used in both file and CCDB mode"};
33  Configurable<std::string> url{"ccdb-url", "http://ccdb-test.cern.ch:8080", "url of the ccdb repository"};
34  Configurable<long> timestamp{"ccdb-timestamp", -1, "timestamp of the object"};
35 
36  void init(o2::framework::InitContext&)
37  {
38  ccdb->setURL(url.value);
39  ccdb->setTimestamp(timestamp.value);
40  ccdb->setCaching(true);
41  ccdb->setLocalObjectValidityChecking();
42  // Not later than now objects
43  ccdb->setCreatedNotAfter(std::chrono::duration_cast<std::chrono::milliseconds>(std::chrono::system_clock::now().time_since_epoch()).count());
44  //
45  const std::string fname = paramfile.value;
46  if (!fname.empty()) { // Loading the parametrization from file
47  resp.LoadParamFromFile(fname.data(), signalname.value, DetectorResponse::kSignal);
48  resp.LoadParamFromFile(fname.data(), sigmaname.value, DetectorResponse::kSigma);
49  } else { // Loading it from CCDB
50  const std::string path = "Analysis/PID/TPC";
51  resp.LoadParam(DetectorResponse::kSignal, ccdb->getForTimeStamp<Parametrization>(path + "/" + signalname.value, timestamp.value));
52  resp.LoadParam(DetectorResponse::kSigma, ccdb->getForTimeStamp<Parametrization>(path + "/" + sigmaname.value, timestamp.value));
53  }
54  }
