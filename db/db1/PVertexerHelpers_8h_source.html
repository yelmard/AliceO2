77 struct TrackVF {
83  enum { kUsed,
84  kNoVtx = -1,
85  kDiscarded = kNoVtx - 1 };
86  float x;
87  float y;
88  float z;
89  float sig2YI;
90  float sig2ZI;
91  float sigYZI;
92  float tgP;
93  float tgL;
94  float cosAlp;
95  float sinAlp;
96 
97  TimeEst timeEst;
98  float wgh = 0.;
99  uint32_t entry = 0;
100  int16_t bin = -1; // seeds histo bin
101  uint8_t srcID = 0;
102  uint8_t flags = 0;
103  int vtxID = kNoVtx;
104 
105  //
106  bool canAssign() const { return wgh > 0. && vtxID == kNoVtx; }
107  bool canUse() const { return vtxID == kNoVtx; }
108  bool canUse(float zmin, float zmax) const
109  {
110  return canUse() && (z > zmin && z < zmax);
111  }
112  bool operator<(const TrackVF& trc) const { return z < trc.z; }
113 
114  float getZForXY(float vx, float vy) const
115  {
116  return z + tgL * (vx * cosAlp + vy * sinAlp - x);
117  }
118 
119  float getResiduals(const PVertex& vtx, float& dy, float& dz) const
120  {
121  // get residuals (Y and Z DCA in track frame) and calculate chi2
122  float dx = vtx.getX() * cosAlp + vtx.getY() * sinAlp - x; // VX rotated to track frame - trackX
123  dy = y + tgP * dx - (-vtx.getX() * sinAlp + vtx.getY() * cosAlp);
124  dz = z + tgL * dx - vtx.getZ();
125  return (dy * dy * sig2YI + dz * dz * sig2ZI) + 2. * dy * dz * sigYZI;
126  }
127 
128  TrackVF() = default;
129  TrackVF(const o2::track::TrackParCov& src, const TimeEst& t_est, uint32_t _entry, uint8_t _srcID)
130  : x(src.getX()), y(src.getY()), z(src.getZ()), tgL(src.getTgl()), tgP(src.getSnp() / std::sqrt(1. - src.getSnp()) * (1. + src.getSnp())), timeEst(t_est), entry(_entry), srcID(_srcID)
131  {
132  o2::utils::sincosf(src.getAlpha(), sinAlp, cosAlp);
133  auto det = src.getSigmaY2() * src.getSigmaZ2() - src.getSigmaZY() * src.getSigmaZY();
134  auto detI = 1. / det;
135  sig2YI = src.getSigmaZ2() * detI;
136  sig2ZI = src.getSigmaY2() * detI;
137  sigYZI = -src.getSigmaZY() * detI;
138  }
139 };

150 struct SeedHisto {
151  float range = 20;
152  float binSize = 0.5;
153  float binSizeInv = 0.;
154  int nFilled = 0;
155  std::vector<int> data;
156 
157  SeedHisto() = delete;
158  SeedHisto(float _range = 20., float _binsize = 0.5) : range(_range), binSize(_binsize)
159  {
160  auto zr = 2 * range;
161  int nzb = zr / binSize;
162  if (nzb * binSize < zr - 1e-9) {
163  nzb++;
164  }
165  binSizeInv = 1. / binSize;
166  range = nzb * binSize / 2.;
167  data.resize(nzb);
168  }
169 
170  int size() const { return data.size(); }
171 
172  void fill(float z)
173  {
174  incrementBin(findBin(z));
175  }
176 
177  void incrementBin(int bin)
178  {
179  data[bin]++;
180  nFilled++;
181  }
182 
183  void decrementBin(int bin)
184  {
185  data[bin]--;
186  nFilled--;
187  }
188 
189  int findBin(float z)
190  {
191  auto d = z + range;
192  if (d < 0.) {
193  return 0;
194  }
195  uint32_t n = d * binSizeInv;
196  return n < data.size() ? n : data.size() - 1;
197  }
198 
199  int findHighestPeakBin() const
200  {
201  if (nFilled < 2) {
202  return -1;
203  }
204  int n = data.size(), maxBin = -1, maxv = 0;
205  for (int i = 0; i < n; i++) {
206  if (data[i] > maxv) {
207  maxv = data[(maxBin = i)];
208  }
209  }
210  return maxBin;
211  }
212 
213  bool isValidBin(int ib) const
214  {
215  return static_cast<uint32_t>(ib) < data.size();
216  }
217 
218  float getBinCenter(int ib) const
219  {
220  return (ib + 0.5) * binSize - range; // no check for being in the range!!!
221  }
222 
223  void discardBin(int ib)
224  { // no check for being in the range!!!
225  nFilled -= data[ib];
226  data[ib] = 0;
227  }
228 };

230 struct TimeCluster {
231  TimeEst timeEst;
232  int first = -1;
233  int last = -1;
234  int count = 0;
235 
236  void clear()
237  {
238  first = last = -1;
239  count = 0;
240  }
241 
242  void addTrack(int i, const TimeEst& trcT)
243  {
244  auto trcTErr2 = trcT.getTimeStampError() * trcT.getTimeStampError();
245  auto trcTErr2Inv = 1. / trcTErr2;
246  if (first < 0) {
247  first = last = i;
248  timeEst.setTimeStamp(trcT.getTimeStamp());
249  timeEst.setTimeStampError(trcT.getTimeStampError());
250  } else {
251  auto vtxTErr2Inv = 1. / (timeEst.getTimeStampError() * timeEst.getTimeStampError());
252  auto vtxTErr2UpdInv = trcTErr2Inv + vtxTErr2Inv;
253  auto vtxTErr2Upd = 1. / vtxTErr2UpdInv;
254  timeEst.setTimeStamp((timeEst.getTimeStamp() * vtxTErr2Inv + trcT.getTimeStamp() * trcTErr2Inv) * vtxTErr2Upd);
255  timeEst.setTimeStampError(std::sqrt(vtxTErr2Upd));
256  if (i > last) {
257  last = i;
258  }
259  }
260  count++;
261  }
262 
263  bool isCompatible(const TimeEst& c, float margin, float cut) const
264  {
265  if (first < 0) {
266  return true;
267  }
268  float dt = timeEst.getTimeStamp() - c.getTimeStamp();
269  if (c.getTimeStampError() && timeEst.getTimeStampError()) {
270  float trcTErr2 = c.getTimeStampError() * c.getTimeStampError();
271  float err = trcTErr2 + timeEst.getTimeStampError() + margin;
272  return dt * dt / err < cut;
273  } else {
274  return std::abs(dt) < cut;
275  }
276  }
277 
278  void merge(TimeCluster& c)
279  {
280  if (c.first < last) {
281  first = c.first;
282  } else {
283  last = c.last;
284  }
285  if (timeEst.getTimeStampError() && c.timeEst.getTimeStampError()) { // weighted average
286  auto cTErr2 = c.timeEst.getTimeStampError() * c.timeEst.getTimeStampError();
287  auto cTErr2Inv = 1. / cTErr2;
288 
289  auto tErr2 = timeEst.getTimeStampError();
290  auto tErr2Inv = 1. / (tErr2 * tErr2);
291  auto tErr2UpdInv = cTErr2Inv + tErr2Inv;
292  auto tErr2Upd = 1. / tErr2UpdInv;
293  timeEst.setTimeStamp((timeEst.getTimeStamp() * tErr2Inv + c.timeEst.getTimeStamp() * cTErr2Inv) * tErr2Upd);
294  timeEst.setTimeStampError(std::sqrt(tErr2Upd));
295  } else {
296  timeEst.setTimeStamp((timeEst.getTimeStamp() * count + c.timeEst.getTimeStamp() * c.count) / (count + c.count));
297  count += c.count;
298  c.count = 0;
299  }
300  }
301 };
