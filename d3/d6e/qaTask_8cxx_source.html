43 struct QATrackingResolution {
44  OutputObj<TH1F> etaDiff{TH1F("etaDiff", ";eta_{MC} - eta_{Rec}", 100, -2, 2)};
45  OutputObj<TH1F> phiDiff{TH1F("phiDiff", ";phi_{MC} - phi_{Rec}", 100, -M_PI, M_PI)};
46  OutputObj<TH1F> ptDiff{TH1F("ptDiff", ";p_{T}_{MC} - p_{T}_{Rec}", 400, -2., 2.)};
47  OutputObj<TH1F> ptRes{TH1F("ptDRes", ";p_{T}_{MC} - p_{T}_{Rec} / p_{T}_{Rec} ", 400, -2., 2.)};
48  OutputObj<TH2F> ptResvspt{TH2F("ptDResvspt", ";p_{T};Res p_{T}", 100, 0., 10., 400, -2., 2.)};
49  OutputObj<TH2F> ptResvseta{TH2F("ptDResvseta", ";#eta;Res p_{T}", 400, -4., 4., 400, -2., 2.)};
50 
51  void process(soa::Join<aod::Collisions, aod::McCollisionLabels>::iterator const& collision, soa::Join<aod::Tracks, aod::McTrackLabels> const& tracks, aod::McParticles const& mcParticles, aod::McCollisions const& mcCollisions)
52  {
53  LOGF(info, "vtx-z (data) = %f | vtx-z (MC) = %f", collision.posZ(), collision.label().posZ());
54  for (auto& track : tracks) {
55  ptDiff->Fill(track.label().pt() - track.pt());
56  ptRes->Fill((track.label().pt() - track.pt()) / track.pt());
57  ptResvspt->Fill(track.pt(), abs(track.label().pt() - track.pt()) / track.pt());
58  ptResvseta->Fill(track.eta(), abs(track.label().pt() - track.pt()) / track.pt());
59  etaDiff->Fill(track.label().eta() - track.eta());
60 
61  auto delta = track.label().phi() - track.phi();
62  if (delta > M_PI)
63  delta -= 2 * M_PI;
64  if (delta < -M_PI)
65  delta += 2 * M_PI;
66  phiDiff->Fill(delta);
67  }
68  }
69 };
