240  std::vector<char> to_vector(bool tcm)
241  {
242  constexpr int CRUWordSize = 16;
243  const char padding[CRUWordSize] = {0};
244 
245  std::vector<char> result(size() * CRUWordSize);
246  char* out = result.data();
247  if (!tcm) {
248  std::memcpy(out, &mEventHeader, EventHeader::PayloadSize);
249  out += EventHeader::PayloadSize;
250  LOG(DEBUG) << "write header words " << (int)mEventHeader.nGBTWords << " orbit " << int(mEventHeader.orbit) << " bc " << int(mEventHeader.bc) << " out " << result.size();
251  if (mIsPadded) {
252  out += CRUWordSize - EventHeader::PayloadSize;
253  }
254  for (int i = 0; i < mEventHeader.nGBTWords; ++i) {
255  std::memcpy(out, &mEventData[2 * i], EventData::PayloadSizeFirstWord);
256  LOG(DEBUG) << " 1st word " << mEventData[2 * i].channelID << " charge " << mEventData[2 * i].charge << " time " << mEventData[2 * i].time << " out " << result.size();
257  out += EventData::PayloadSizeFirstWord;
258  std::memcpy(out, &mEventData[2 * i + 1], EventData::PayloadSizeSecondWord);
259  out += EventData::PayloadSizeSecondWord;
260  LOG(DEBUG) << " 2nd word " << mEventData[2 * i + 1].channelID << " charge " << mEventData[2 * i + 1].charge << " time " << mEventData[2 * i + 1].time << " out " << result.size();
261  if (mIsPadded) {
262  out += CRUWordSize - EventData::PayloadSizeSecondWord - EventData::PayloadSizeFirstWord;
263  }
264  }
265  } else {
266  // TCM data
267  std::memcpy(out, &mEventHeader, EventHeader::PayloadSize);
268  out += EventHeader::PayloadSize;
269  LOG(DEBUG) << "write TCM header words " << (int)mEventHeader.nGBTWords << " orbit " << int(mEventHeader.orbit) << " bc " << int(mEventHeader.bc) << " out " << result.size();
270  std::memcpy(out, &mTCMdata, sizeof(TCMdata));
271  out += sizeof(TCMdata);
272  LOG(DEBUG) << "write TCM words " << sizeof(mTCMdata) << " orbit " << int(mEventHeader.orbit) << " bc " << int(mEventHeader.bc) << " out " << result.size() << " sum time A " << mTCMdata.timeA;
273  }
274 
275  return result;
276  }

292 class DataPageWriter
293 {
294  std::vector<char> mBuffer;
295  int mNpacketsInBuffer = 0;
296  std::vector<std::vector<char>> mPages;
297  std::vector<int> mNpackets;
298  static constexpr int MAX_Page_size = 8192;
299 
300  public:
301  o2::header::RAWDataHeader mRDH;
302  void flush(std::ostream& str)
303  {
304  writePage();
305  mRDH.stop = 0;
306  for (int page = 0; page < int(mPages.size()); ++page) {
307  mRDH.memorySize = mPages[page].size() + mRDH.headerSize;
308  mRDH.offsetToNext = mRDH.memorySize;
309  mRDH.packetCounter = mNpackets[page];
310  str.write(reinterpret_cast<const char*>(&mRDH), sizeof(mRDH));
311  str.write(mPages[page].data(), mPages[page].size());
312  mRDH.pageCnt++;
313  }
314  if (!mPages.empty()) {
315  mRDH.memorySize = mRDH.headerSize;
316  mRDH.offsetToNext = mRDH.memorySize;
317  mRDH.stop = 1;
318  mRDH.pageCnt++;
319  str.write(reinterpret_cast<const char*>(&mRDH), sizeof(mRDH));
320  mPages.clear();
321  mNpackets.clear();
322  }
323  }
324 
325  void writePage()
326  {
327  if (mBuffer.size() == 0)
328  return;
329  mPages.emplace_back(std::move(mBuffer));
330  LOG(DEBUG) << " writePage " << mBuffer.size();
331  mNpackets.push_back(mNpacketsInBuffer);
332  mNpacketsInBuffer = 0;
333  mBuffer.clear();
334  }
335 
336  void write(std::vector<char> const& new_data)
337  {
338  if (mBuffer.size() + new_data.size() + mRDH.headerSize > MAX_Page_size) {
339  LOG(DEBUG) << " write rest " << mBuffer.size() << " " << new_data.size() << " " << mRDH.headerSize;
340  writePage();
341  }
342  LOG(DEBUG) << " write vector " << new_data.size() << " buffer " << mBuffer.size() << " RDH " << mRDH.headerSize << " new data " << new_data.data();
343  mBuffer.insert(mBuffer.end(), new_data.begin(), new_data.end());
344  mNpacketsInBuffer++;
345  LOG(DEBUG) << " write vector end mBuffer.size " << mBuffer.size() << " mNpacketsInBuffer " << mNpacketsInBuffer << " newdtata " << new_data.size();
346  }
