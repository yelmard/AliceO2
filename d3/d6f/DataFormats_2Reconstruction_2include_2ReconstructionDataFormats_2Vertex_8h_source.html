101 // Base primary vertex class, with position, error, N candidates and flags field
102 // The Stamp template parameter allows to define vertex (time)stamp in different
103 // formats (ITS ROFrame ID, real time + error etc)
104 
105 template <typename Stamp>
106 class Vertex : public VertexBase
107 {
108  public:
109  using ushort = unsigned short;
110  enum Flags : ushort {
111  TimeValidated = 0x1 << 0, // Flag that the vertex was validated by external time measurement (e.g. FIT)
112  FlagsMask = 0xffff
113  };
114 
115  Vertex() = default;
116  ~Vertex() = default;
117  Vertex(const Point3D<float>& pos, const std::array<float, kNCov>& cov, ushort nCont, float chi2)
118  : VertexBase(pos, cov), mNContributors(nCont), mChi2(chi2)
119  {
120  }
121 
122  ushort getNContributors() const { return mNContributors; }
123  void setNContributors(ushort v) { mNContributors = v; }
124  void addContributor() { mNContributors++; }
125 
126  ushort getFlags() const { return mBits; }
127  bool isFlagSet(uint f) const { return mBits & (FlagsMask & f); }
128  void setFlags(ushort f) { mBits |= FlagsMask & f; }
129  void resetFrags(ushort f = FlagsMask) { mBits &= ~(FlagsMask & f); }
130 
131  void setChi2(float v) { mChi2 = v; }
132  float getChi2() const { return mChi2; }
133 
134  const Stamp& getTimeStamp() const { return mTimeStamp; }
135  Stamp& getTimeStamp() { return mTimeStamp; }
136  void setTimeStamp(const Stamp& v) { mTimeStamp = v; }
137 
138  protected:
139  float mChi2 = 0;
140  ushort mNContributors = 0;
141  ushort mBits = 0;
142  Stamp mTimeStamp;
143 
144  ClassDefNV(Vertex, 3);
145 };
