41 void VertexTrackMatcher::process(const gsl::span<const PVertex>& vertices,
42  const gsl::span<const GIndex>& v2tfitIDs,
43  const gsl::span<const VRef>& v2tfitRefs,
44  const gsl::span<const TrackTPCITS>& tpcitsTracks,
45  const gsl::span<const TrackITS>& itsTracks,
46  const gsl::span<const ITSROFR>& itsROFR,
47  const gsl::span<const TrackTPC>& tpcTracks,
48  std::vector<GIndex>& trackIndex,
49  std::vector<VRef>& vtxRefs)
50 {
51 
52  TmpMap tmpMap;
53  tmpMap.reserve(vertices.size());
54 
55  // TPC/ITS and TPC tracks are not sorted in time, do this and exclude indiced of tracks used in the vertex fit
56  std::vector<int> idTPCITS(tpcitsTracks.size()); // indices of TPCITS tracks sorted in time
57  std::vector<int> idVtxIRMin(vertices.size()); // indices of vertices sorted in IRmin
58  std::vector<int> flgITS(itsTracks.size(), 0);
59  std::vector<int> idTPC(tpcTracks.size()); // indices of TPC tracks sorted in min time
60  std::iota(idTPC.begin(), idTPC.end(), 0);
61  std::iota(idTPCITS.begin(), idTPCITS.end(), 0);
62  std::iota(idVtxIRMin.begin(), idVtxIRMin.end(), 0);
63  for (auto id : v2tfitIDs) { // flag matched ITS-TPC tracks used in the vertex fit, we exclude them from this association
64  idTPCITS[id.getIndex()] = -1;
65  }
66  for (const auto& tpcits : tpcitsTracks) { // flag standalone ITS and TPC tracks used in global matched, we exclude them from association to vertex
67  flgITS[tpcits.getRefITS()] = -1;
68  idTPC[tpcits.getRefTPC()] = -1;
69  }
70  std::sort(idVtxIRMin.begin(), idVtxIRMin.end(), [&vertices](int i, int j) { // sort according to IRMin
71  return vertices[i].getIRMin() < vertices[j].getIRMin();
72  });
73  std::sort(idTPCITS.begin(), idTPCITS.end(), [&tpcitsTracks](int i, int j) { // sort according to central time
74  float tI = (i < 0) ? 1e9 : tpcitsTracks[i].getTimeMUS().getTimeStamp();
75  float tJ = (j < 0) ? 1e9 : tpcitsTracks[j].getTimeMUS().getTimeStamp();
76  return tI < tJ;
77  });
78 
79  std::vector<TBracket> tpcTimes;
80  tpcTimes.reserve(tpcTracks.size());
81  for (const auto& trc : tpcTracks) {
82  tpcTimes.emplace_back(tpcTimeBin2MUS(trc.getTime0() - trc.getDeltaTBwd()), tpcTimeBin2MUS(trc.getTime0() + trc.getDeltaTFwd()));
83  }
84  std::sort(idTPC.begin(), idTPC.end(), [&tpcTimes](int i, int j) { // sort according to max time
85  float tI = (i < 0) ? 1e9 : tpcTimes[i].max();
86  float tJ = (j < 0) ? 1e9 : tpcTimes[j].max();
87  return tI < tJ;
88  });
89 
90  // Important: do this in the same order in which VtxTrackIndex::Source enums are defined!
91  attachTPCITS(tmpMap, tpcitsTracks, idTPCITS, vertices);
92  attachITS(tmpMap, itsTracks, itsROFR, flgITS, vertices, idVtxIRMin);
93  attachTPC(tmpMap, tpcTimes, idTPC, vertices, idVtxIRMin);
94 
95  // build vector of global indices
96  trackIndex.clear();
97  vtxRefs.clear();
98  memset(idTPCITS.data(), 0, sizeof(int) * idTPCITS.size());
99  memset(idTPC.data(), 0, sizeof(int) * idTPC.size());
100  memset(flgITS.data(), 0, sizeof(int) * flgITS.size());
101  std::array<std::vector<int>*, GIndex::NSources> vptr;
102  vptr[GIndex::TPCITS] = &idTPCITS;
103  vptr[GIndex::ITS] = &flgITS;
104  vptr[GIndex::TPC] = &idTPC;
105  int nv = vertices.size();
106  // flag tracks attached to >1 vertex
107  for (int iv = 0; iv < nv; iv++) {
108  const auto& trvec = tmpMap[iv];
109  for (auto gid : trvec) {
110  (*vptr[gid.getSource()])[gid.getIndex()]++;
111  }
112  }
113 
114  for (int iv = 0; iv < nv; iv++) {
115  int srcStart[GIndex::NSources + 1];
116  int entry = trackIndex.size();
117  srcStart[GIndex::TPCITS] = entry;
118  for (int is = 1; is < GIndex::NSources; is++) {
119  srcStart[is] = -1;
120  }
121 
122  // 1st: attach indices of global tracks used in vertex fit
123  int idMin = v2tfitRefs[iv].getFirstEntry(), idMax = idMin + v2tfitRefs[iv].getEntries();
124  for (int id = idMin; id < idMax; id++) {
125  trackIndex.push_back(v2tfitIDs[id]);
126  }
127 
128  // 2nd: attach non-contributing tracks
129  const auto& trvec = tmpMap[iv];
130  for (const auto gid0 : trvec) {
131  int src = gid0.getSource();
132  if (srcStart[src] == -1) {
133  srcStart[src] = trackIndex.size();
134  }
135  auto& gid = trackIndex.emplace_back(gid0);
136  if ((*vptr[src])[gid.getIndex()] > 1) {
137  gid.setBit(GIndex::Ambiguous);
138  }
139  }
140 
141  auto& vr = vtxRefs.emplace_back(entry, trackIndex.size() - entry);
142  srcStart[GIndex::NSources] = trackIndex.size();
143  for (int is = GIndex::NSources; is > 0; is--) {
144  if (srcStart[is] == -1) { // in case the source did not contribute
145  srcStart[is] = srcStart[is + 1];
146  }
147  vr.setFirstEntryOfSource(is, srcStart[is]);
148  }
149  }
150 }

153 void VertexTrackMatcher::attachTPCITS(TmpMap& tmpMap, const gsl::span<const TrackTPCITS>& tpcits, const std::vector<int>& idTPCITS, const gsl::span<const PVertex>& vertices)
154 {
155  int itrCurr = 0, nvt = vertices.size(), ntr = tpcits.size();
156  // indices of tracks used for vertex fit will be in the end, find their start and max error
157  float maxErr = 0;
158  for (int i = 0; i < ntr; i++) {
159  if (idTPCITS[i] < 0) {
160  ntr = i;
161  break;
162  }
163  float err = tpcits[idTPCITS[i]].getTimeMUS().getTimeStampError();
164  if (maxErr < err) {
165  maxErr = err;
166  }
167  }
168  auto maxErr2 = maxErr * maxErr;
169  for (int ivtCurr = 0; ivtCurr < nvt; ivtCurr++) {
170  const auto& vtxT = vertices[ivtCurr].getTimeStamp();
171  auto rangeT = mPVParams->nSigmaTimeCut * std::sqrt(maxErr2 + vtxT.getTimeStampError() * vtxT.getTimeStampError());
172  float minTime = vtxT.getTimeStamp() - rangeT, maxTime = vtxT.getTimeStamp() + rangeT;
173  // proceed to the 1st track having compatible time
174  int itr = itrCurr;
175  while (itr < ntr) {
176  const auto& trcT = tpcits[idTPCITS[itr]].getTimeMUS();
177  if (trcT.getTimeStamp() < minTime) {
178  itrCurr = itr;
179  } else if (trcT.getTimeStamp() > maxTime) {
180  break;
181  } else if (compatibleTimes(vtxT, trcT)) {
182  tmpMap[ivtCurr].emplace_back(idTPCITS[itr], GIndex::TPCITS);
183  }
184  itr++;
185  }
186  }
187 }

190 void VertexTrackMatcher::attachITS(TmpMap& tmpMap, const gsl::span<const TrackITS>& itsTracks, const gsl::span<const ITSROFR>& itsROFR, const std::vector<int>& flITS,
191  const gsl::span<const PVertex>& vertices, std::vector<int>& idxVtx)
192 {
193  int irofCurr = 0, nvt = vertices.size(), ntr = itsTracks.size();
194  int nROFs = itsROFR.size();
195 
196  for (int ivtCurr = 0; ivtCurr < nvt; ivtCurr++) {
197  const auto& vtx = vertices[idxVtx[ivtCurr]]; // we iterate over vertices in order of their getIRMin()
198  int irof = irofCurr;
199  while (irof < nROFs) {
200  const auto& rofr = itsROFR[irof];
201  auto irMin = rofr.getBCData(), irMax = irMin + mITSROFrameLengthInBC;
202  if (irMax < vtx.getIRMin()) {
203  irofCurr = irof;
204  } else if (irMin > vtx.getIRMax()) {
205  break;
206  } else {
207  if (irMax == vtx.getIRMin()) {
208  irofCurr = irof;
209  }
210  int maxItr = rofr.getNEntries() + rofr.getFirstEntry();
211  for (int itr = rofr.getFirstEntry(); itr < maxItr; itr++) {
212  if (flITS[itr] != -1) {
213  tmpMap[ivtCurr].emplace_back(itr, GIndex::ITS);
214  }
215  }
216  }
217  irof++;
218  }
219  }
220 }

223 void VertexTrackMatcher::attachTPC(TmpMap& tmpMap, const std::vector<TBracket>& tpcTimes, const std::vector<int>& idTPC,
224  const gsl::span<const PVertex>& vertices, std::vector<int>& idVtx)
225 {
226  int itrCurr = 0, nvt = vertices.size(), ntr = idTPC.size();
227  // indices of tracks used for vertex fit will be in the end, find their start and max error
228  for (int i = 0; i < ntr; i++) {
229  if (idTPC[i] < 0) {
230  ntr = i;
231  break;
232  }
233  }
234  for (int ivtCurr = 0; ivtCurr < nvt; ivtCurr++) {
235  const auto& vtx = vertices[idVtx[ivtCurr]];
236  TBracket tV(vtx.getIRMin().differenceInBC(mStartIR) * o2::constants::lhc::LHCBunchSpacingNS * 1e-3,
237  vtx.getIRMax().differenceInBC(mStartIR) * o2::constants::lhc::LHCBunchSpacingNS * 1e-3);
238  // proceed to the 1st track having compatible time, idTPC provides track IDs ordered in max time
239  int itr = itrCurr;
240  while (itr < ntr) {
241  const auto& tT = tpcTimes[idTPC[itr]];
242  auto rel = tV.isOutside(tT);
243  if (rel == TBracket::Below) { // tmax of track < tmin of vtx
244  itrCurr = itr;
245  } else if (rel == TBracket::Inside) {
246  tmpMap[ivtCurr].emplace_back(idTPC[itr], GIndex::TPC);
247  } else if (tT.max() - mMaxTPCDriftTimeMUS > tV.max()) {
248  break;
249  }
250  itr++;
251  }
252  }
253 }

262 bool VertexTrackMatcher::compatibleTimes(const TimeEst& vtxT, const TimeEst& trcT) const
263 {
264  float err2 = vtxT.getTimeStampError() * vtxT.getTimeStampError() + trcT.getTimeStampError() * trcT.getTimeStampError();
265  float dfred = (vtxT.getTimeStamp() - trcT.getTimeStamp()) * mPVParams->nSigmaTimeCut;
266  return dfred * dfred < err2;
267 }
