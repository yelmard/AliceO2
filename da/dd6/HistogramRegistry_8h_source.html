128 template <typename T>
129 std::unique_ptr<TH1> createTH1FromSpec(HistogramSpec const& spec)
130 {
131  if (spec.config.axes.size() == 0) {
132  throw std::runtime_error("No arguments available in spec to create a histogram");
133  }
134  if (spec.config.binsEqual) {
135  return std::make_unique<T>(spec.name.data(), spec.readableName.data(), spec.config.axes[0].nBins, spec.config.axes[0].bins[0], spec.config.axes[0].bins[1]);
136  }
137  return std::make_unique<T>(spec.name.data(), spec.readableName.data(), spec.config.axes[0].nBins, spec.config.axes[0].bins.data());
138 }

262 class HistogramRegistry
263 {
264  public:
265  HistogramRegistry(char const* const name_, bool enable, std::vector<HistogramSpec> specs, OutputObjHandlingPolicy policy_ = OutputObjHandlingPolicy::AnalysisObject)
266  : name(name_),
267  policy(policy_),
268  enabled(enable),
269  mRegistryKey(),
270  mRegistryValue(),
271  mHistogramCreationCallbacks({HistogramCallbacks::createTH1D(),
272  HistogramCallbacks::createTH1F(),
273  HistogramCallbacks::createTH1I(),
274  HistogramCallbacks::createTH2D(),
275  HistogramCallbacks::createTH2F(),
276  HistogramCallbacks::createTH2I(),
277  HistogramCallbacks::createTH3D(),
278  HistogramCallbacks::createTH3F(),
279  HistogramCallbacks::createTH3I()})
280  {
281  mRegistryKey.fill(0u);
282  for (auto& spec : specs) {
283  insert(spec);
284  }
285  }

288  auto& get(char const* const name) const
289  {
290  const uint32_t id = compile_time_hash(name);
291  const uint32_t i = imask(id);
292  if (O2_BUILTIN_LIKELY(id == mRegistryKey[i])) {
293  return mRegistryValue[i];
294  }
295  for (auto j = 1u; j < MAX_REGISTRY_SIZE; ++j) {
296  if (id == mRegistryKey[imask(j + i)]) {
297  return mRegistryValue[imask(j + i)];
298  }
299  }
300  throw std::runtime_error("No match found!");
301  }
302 
304  auto& operator()(char const* const name) const
305  {
306  return get(name);
307  }
308 
310  OutputSpec const spec()
311  {
312  ConcreteDataMatcher matcher{"HIST", "\0", 0};
313  strncpy(matcher.description.str, this->name.data(), 16);
314  return OutputSpec{OutputLabel{this->name}, matcher};
315  }
316 
317  OutputRef ref()
318  {
319  return OutputRef{std::string{this->name}, 0, o2::header::Stack{OutputObjHeader{policy, taskHash}}};
320  }
321  void setHash(uint32_t hash)
322  {
323  taskHash = hash;
324  }
325 
326  TFolder* operator*()
327  {
328  TFolder* folder = new TFolder(this->name.c_str(), this->name.c_str());
329  for (auto j = 0u; j < MAX_REGISTRY_SIZE; ++j) {
330  if (mRegistryValue[j].get() != nullptr) {
331  auto hist = mRegistryValue[j].get();
332  folder->Add(hist);
333  }
334  }
335  folder->SetOwner();
336  return folder;
337  }

362  mutable uint32_t lookup = 0;
363 
364  private:
365  void insert(HistogramSpec& spec)
366  {
367  uint32_t i = imask(spec.id);
368  for (auto j = 0u; j < MAX_REGISTRY_SIZE; ++j) {
369  if (mRegistryValue[imask(j + i)].get() == nullptr) {
370  mRegistryKey[imask(j + i)] = spec.id;
371  mRegistryValue[imask(j + i)] = mHistogramCreationCallbacks[spec.config.type](spec);
372  lookup += j;
373  return;
374  }
375  }
376  throw std::runtime_error("Internal array is full.");
377  }
378 
379  inline constexpr uint32_t imask(uint32_t i) const
380  {
381  return i & mask;
382  }
383  std::string name;
384  bool enabled;
385  OutputObjHandlingPolicy policy;
386  uint32_t taskHash;
387 
390  static constexpr uint32_t mask = 0x1FF;
391  static constexpr uint32_t MAX_REGISTRY_SIZE = mask + 1;
392  std::array<uint32_t, MAX_REGISTRY_SIZE> mRegistryKey;
393  std::array<std::unique_ptr<TH1>, MAX_REGISTRY_SIZE> mRegistryValue;
394  std::vector<HistogramCreationCallback> mHistogramCreationCallbacks;
