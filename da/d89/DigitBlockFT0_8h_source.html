56  template <class DataBlockType>
57  //,typename enable_if<boost::mpl::contains<ProccessBlockTypes,DataBlockType>::type::value>::type > //TODO traits
58  void proccess(DataBlockType& dataBlock, int linkID)
59  {
60  if constexpr (std::is_same<DataBlockType, DataBlockPM>::value) { //Filling data from PM
61  for (int iEventData = 0; iEventData < dataBlock.DataBlockWrapper<RawDataPM>::mNelements; iEventData++) {
62  mVecChannelData.emplace_back(uint8_t(sLookupTable.getChannel(linkID, dataBlock.DataBlockWrapper<RawDataPM>::mData[iEventData].channelID)),
63  int(dataBlock.DataBlockWrapper<RawDataPM>::mData[iEventData].time),
64  int(dataBlock.DataBlockWrapper<RawDataPM>::mData[iEventData].charge),
65  dataBlock.DataBlockWrapper<RawDataPM>::mData[iEventData].getFlagWord());
66  }
67  } else if constexpr (std::is_same<DataBlockType, DataBlockTCM>::value) { //Filling data from TCM (normal/extended mode)
68  dataBlock.DataBlockWrapper<RawDataTCM>::mData[0].fillTrigger(mDigit.mTriggers);
69  }
70 
71  else if constexpr (std::is_same<DataBlockType, DataBlockTCMext>::value) { //Filling data from TCM, extended mode. Same proccess as for normal mode, for now.
72  dataBlock.DataBlockWrapper<RawDataTCM>::mData[0].fillTrigger(mDigit.mTriggers);
73  }
74  }
75  void popData(std::vector<Digit>& vecDigits, std::vector<ChannelData>& vecChannelData)
76  {
77  //last digit filling
78  mDigit.ref.set(vecChannelData.size(), mVecChannelData.size());
79  mDigit.mEventID = sEventID;
80  //
81  vecDigits.push_back(std::move(mDigit));
82  std::move(mVecChannelData.begin(), mVecChannelData.end(), std::back_inserter(vecChannelData));
83  mVecChannelData.clear();
84 
85  sEventID++; //Increasing static eventID. After each poping of the data, it will increase
86  }
