29 int PVertexer::process(gsl::span<const o2d::TrackTPCITS> tracksITSTPC, gsl::span<const o2::ft0::RecPoints> ft0Data,
30  std::vector<PVertex>& vertices, std::vector<GIndex>& vertexTrackIDs, std::vector<V2TRef>& v2tRefs)
31 {
32  createTracksPool(tracksITSTPC);
33 
34  //clusterizeTimeBruteForce();
35  clusterizeTime();
36 
37  std::vector<PVertex> verticesLoc;
38  std::vector<int> vertexTrackIDsLoc;
39  std::vector<V2TRef> v2tRefsLoc;
40  std::vector<float> validationTimes;
41 
42  for (auto tc : mTimesClusters) {
43  VertexingInput inp;
44  inp.idRange = gsl::span<int>((int*)&mSortedTrackID[tc.first], tc.count);
45  inp.scaleSigma2 = 3. * estimateScale2();
46  inp.timeEst = tc.timeEst;
47  findVertices(inp, verticesLoc, vertexTrackIDsLoc, v2tRefsLoc);
48  }
49 
50  // sort in time
51  std::vector<int> vtTimeSortID(verticesLoc.size());
52  std::iota(vtTimeSortID.begin(), vtTimeSortID.end(), 0);
53  std::sort(vtTimeSortID.begin(), vtTimeSortID.end(), [&verticesLoc](int i, int j) {
54  return verticesLoc[i].getTimeStamp().getTimeStamp() < verticesLoc[j].getTimeStamp().getTimeStamp();
55  });
56 
57  vertices.clear();
58  v2tRefs.clear();
59  vertexTrackIDs.clear();
60  vertices.reserve(verticesLoc.size());
61  v2tRefs.reserve(v2tRefsLoc.size());
62  vertexTrackIDs.reserve(vertexTrackIDsLoc.size());
63 
64  int trCopied = 0, count = 0, vtimeID = 0;
65  for (auto i : vtTimeSortID) {
66  auto& vtx = verticesLoc[i];
67 
68  bool irSet = setCompatibleIR(vtx);
69  if (!irSet) {
70  continue;
71  }
72  // do we need to validate by FT0 ?
73  if (mValidateWithFT0) {
74  auto bestMatch = getBestFT0Trigger(vtx, ft0Data, vtimeID);
75  if (bestMatch.first >= 0) {
76  vtx.setFlags(PVertex::TimeValidated);
77  if (bestMatch.second == 1) {
78  vtx.setIR(ft0Data[bestMatch.first].getInteractionRecord());
79  }
80  LOG(DEBUG) << "Validated with t0 " << bestMatch.first << " with " << bestMatch.second << " candidates";
81  } else if (vtx.getNContributors() >= mPVParams->minNContributorsForFT0cut) {
82  LOG(DEBUG) << "Discarding " << vtx;
83  continue; // reject
84  }
85  }
86  vertices.push_back(vtx);
87  int it = v2tRefsLoc[i].getFirstEntry(), itEnd = it + v2tRefsLoc[i].getEntries(), dest0 = vertexTrackIDs.size();
88  for (; it < itEnd; it++) {
89  auto& gid = vertexTrackIDs.emplace_back(vertexTrackIDsLoc[it], GIndex::TPCITS);
90  gid.setBit(GIndex::Contributor);
91  }
92  v2tRefs.emplace_back(dest0, v2tRefsLoc[i].getEntries());
93  LOG(DEBUG) << "#" << count++ << " " << vertices.back() << " | " << v2tRefs.back().getEntries() << " indices from " << v2tRefs.back().getFirstEntry(); // RS REM
94  }
95 
96  return vertices.size();
97 }

112 int PVertexer::findVertices(const VertexingInput& input, std::vector<PVertex>& vertices, std::vector<int>& vertexTrackIDs, std::vector<V2TRef>& v2tRefs)
113 {
114  // find vertices using tracks with indices (sorted in time) from idRange from "tracks" pool. The pool may containt arbitrary number of tracks,
115  // only those which are in the idRange and have canUse()==true, will be used.
116  // Results are placed in vertices and v2tRefs vectors
117  int nfound = 0, ntr = input.idRange.size();
118  if (ntr < mPVParams->minTracksPerVtx) {
119  return nfound;
120  }
121  //
122  SeedHisto seedHisto(mPVParams->zHistoRange, mPVParams->zHistoBinSize);
123 
124  for (int i : input.idRange) {
125  if (mTracksPool[i].canUse()) {
126  mTracksPool[i].bin = seedHisto.findBin(mTracksPool[i].getZForXY(mMeanVertex.getX(), mMeanVertex.getY()));
127  seedHisto.incrementBin(mTracksPool[i].bin);
128  }
129  }
130  if (seedHisto.nFilled < mPVParams->minTracksPerVtx) {
131  return nfound;
132  }
133 
134  while (1) {
135  int peakBin = seedHisto.findHighestPeakBin(); // find next seed
136  if (!seedHisto.isValidBin(peakBin)) {
137  break;
138  }
139  float zv = seedHisto.getBinCenter(peakBin);
140  LOG(DEBUG) << "Seeding with Z=" << zv << " bin " << peakBin;
141  PVertex vtx;
142  vtx.setXYZ(mMeanVertex.getX(), mMeanVertex.getY(), zv);
143  vtx.setTimeStamp(input.timeEst);
144  if (findVertex(input, vtx)) {
145  finalizeVertex(input, vtx, vertices, v2tRefs, vertexTrackIDs, seedHisto);
146  nfound++;
147  } else { // suppress failed seeding bin and its proximities
148  auto delta = std::sqrt(vtx.getChi2()) * mStatZErr.getMean() * getTukey(); // largest scale used will be transferred as chi2
149  int proximity = delta * seedHisto.binSizeInv;
150  int bmin = std::max(0, peakBin - proximity), bmax = std::min(peakBin + proximity + 1, seedHisto.size());
151  LOG(DEBUG) << "suppress bins for delta=" << delta << " (" << std::sqrt(vtx.getChi2()) << "*" << mStatZErr.getMean() << "*" << getTukey() << ")"
152  << " bins " << bmin << " : " << bmax - 1;
153  for (int i = bmin; i < bmax; i++) {
154  seedHisto.discardBin(i);
155  }
156  }
157  }
158  return nfound;
159 }

161 //______________________________________________
162 bool PVertexer::findVertex(const VertexingInput& input, PVertex& vtx)
163 {
164  // fit vertex taking provided vertex as a seed
165  // tracks pool may contain arbitrary number of tracks, only those which are in
166  // the idRange (indices of tracks sorted in time) will be used.
167 
168  VertexSeed vtxSeed(vtx, input.useConstraint, input.fillErrors);
169  vtxSeed.setScale(input.scaleSigma2, mTukey2I);
170  vtxSeed.scaleSigma2Prev = input.scaleSigma2;
171  // vtxSeed.setTimeStamp( timeEstimate(input) );
172  LOG(DEBUG) << "Start time guess: " << vtxSeed.getTimeStamp();
173  vtx.setChi2(1.e30);
174  //
175  FitStatus result = FitStatus::IterateFurther;
176  bool found = false;
177  while (result == FitStatus::IterateFurther) {
178  vtxSeed.resetForNewIteration();
179  vtxSeed.nIterations++;
180  LOG(DEBUG) << "iter " << vtxSeed.nIterations << " with scale=" << vtxSeed.scaleSigma2 << " prevScale=" << vtxSeed.scaleSigma2Prev;
181  result = fitIteration(input, vtxSeed);
182 
183  if (result == FitStatus::OK) {
184  result = evalIterations(vtxSeed, vtx);
185  } else if (result == FitStatus::NotEnoughTracks) {
186  if (vtxSeed.nIterations <= mPVParams->maxIterations && upscaleSigma(vtxSeed)) {
187  LOG(DEBUG) << "Upscaling scale to " << vtxSeed.scaleSigma2;
188  result = FitStatus::IterateFurther;
189  continue; // redo with stronger rescaling
190  } else {
191  break;
192  }
193  } else if (result == FitStatus::PoolEmpty || result == FitStatus::Failure) {
194  break;
195  } else {
196  LOG(FATAL) << "Unknown fit status " << int(result);
197  }
198  }
199  LOG(DEBUG) << "Stopped with scale=" << vtxSeed.scaleSigma2 << " prevScale=" << vtxSeed.scaleSigma2Prev << " result = " << int(result);
200 
201  if (result != FitStatus::OK) {
202  vtx.setChi2(vtxSeed.maxScaleSigma2Tested);
203  return false;
204  } else {
205  return true;
206  }
207 }

209 //___________________________________________________________________
210 PVertexer::FitStatus PVertexer::fitIteration(const VertexingInput& input, VertexSeed& vtxSeed)
211 {
212  int nTested = 0;
213  for (int i : input.idRange) {
214  if (mTracksPool[i].canUse()) {
215  nTested++;
216  accountTrack(mTracksPool[i], vtxSeed);
217  }
218  }
219  vtxSeed.maxScaleSigma2Tested = vtxSeed.scaleSigma2;
220  if (vtxSeed.getNContributors() < mPVParams->minTracksPerVtx) {
221  return nTested < mPVParams->minTracksPerVtx ? FitStatus::PoolEmpty : FitStatus::NotEnoughTracks;
222  }
223  if (vtxSeed.useConstraint) {
224  applyConstraint(vtxSeed);
225  }
226  if (!solveVertex(vtxSeed)) {
227  return FitStatus::Failure;
228  }
229  return FitStatus::OK;
230 }

232 //___________________________________________________________________
233 void PVertexer::accountTrack(TrackVF& trc, VertexSeed& vtxSeed) const
234 {
235  // deltas defined as track - vertex
236  float dt, trErr2I = 0, dy, dz, chi2T = trc.getResiduals(vtxSeed, dy, dz); // track-vertex residuals and chi2
237  auto& timeV = vtxSeed.getTimeStamp();
238  auto& timeT = trc.timeEst;
239  float ndff = 1. / 2;
240  bool noTime = false;
241  if (timeV.getTimeStampError() < 0.) {
242  noTime = true;
243  } else {
244  dt = timeT.getTimeStamp() - timeV.getTimeStamp();
245  trErr2I = 1. / (timeT.getTimeStampError() * timeT.getTimeStampError());
246  if (mPVParams->useTimeInChi2) {
247  chi2T += dt * dt * trErr2I;
248  ndff = 1. / 3.;
249  }
250  }
251  chi2T *= ndff;
252  float wghT = (1.f - chi2T * vtxSeed.scaleSig2ITuk2I); // weighted distance to vertex
253  if (wghT < kAlmost0F) {
254  trc.wgh = 0.f;
255  return;
256  }
257  float syyI(trc.sig2YI), szzI(trc.sig2ZI), syzI(trc.sigYZI);
258 
259  //
260  vtxSeed.wghSum += wghT;
261  vtxSeed.wghChi2 += wghT * chi2T;
262  //
263  syyI *= wghT;
264  syzI *= wghT;
265  szzI *= wghT;
266  trc.wgh = wghT;
267  //
268  // aux variables
269  double tmpSP = trc.sinAlp * trc.tgP, tmpCP = trc.cosAlp * trc.tgP,
270  tmpSC = trc.sinAlp + tmpCP, tmpCS = -trc.cosAlp + tmpSP,
271  tmpCL = trc.cosAlp * trc.tgL, tmpSL = trc.sinAlp * trc.tgL,
272  tmpYXP = trc.y - trc.tgP * trc.x, tmpZXL = trc.z - trc.tgL * trc.x,
273  tmpCLzz = tmpCL * szzI, tmpSLzz = tmpSL * szzI, tmpSCyz = tmpSC * syzI,
274  tmpCSyz = tmpCS * syzI, tmpCSyy = tmpCS * syyI, tmpSCyy = tmpSC * syyI,
275  tmpSLyz = tmpSL * syzI, tmpCLyz = tmpCL * syzI;
276  //
277  // symmetric matrix equation
278  vtxSeed.cxx += tmpCL * (tmpCLzz + tmpSCyz + tmpSCyz) + tmpSC * tmpSCyy; // dchi^2/dx/dx
279  vtxSeed.cxy += tmpCL * (tmpSLzz + tmpCSyz) + tmpSL * tmpSCyz + tmpSC * tmpCSyy; // dchi^2/dx/dy
280  vtxSeed.cxz += -trc.sinAlp * syzI - tmpCLzz - tmpCP * syzI; // dchi^2/dx/dz
281  vtxSeed.cx0 += -(tmpCLyz + tmpSCyy) * tmpYXP - (tmpCLzz + tmpSCyz) * tmpZXL; // RHS
282  //
283  vtxSeed.cyy += tmpSL * (tmpSLzz + tmpCSyz + tmpCSyz) + tmpCS * tmpCSyy; // dchi^2/dy/dy
284  vtxSeed.cyz += -(tmpCSyz + tmpSLzz); // dchi^2/dy/dz
285  vtxSeed.cy0 += -tmpYXP * (tmpCSyy + tmpSLyz) - tmpZXL * (tmpCSyz + tmpSLzz); // RHS
286  //
287  vtxSeed.czz += szzI; // dchi^2/dz/dz
288  vtxSeed.cz0 += tmpZXL * szzI + tmpYXP * syzI; // RHS
289  //
290  if (!noTime) {
291  trErr2I *= wghT;
292  vtxSeed.tMeanAcc += timeT.getTimeStamp() * trErr2I;
293  vtxSeed.tMeanAccErr += trErr2I;
294  }
295  vtxSeed.addContributor();
296 }

299 bool PVertexer::solveVertex(VertexSeed& vtxSeed) const
300 {
301  ROOT::Math::SMatrix<double, 3, 3, ROOT::Math::MatRepSym<double, 3>> mat;
302  mat(0, 0) = vtxSeed.cxx;
303  mat(0, 1) = vtxSeed.cxy;
304  mat(0, 2) = vtxSeed.cxz;
305  mat(1, 1) = vtxSeed.cyy;
306  mat(1, 2) = vtxSeed.cyz;
307  mat(2, 2) = vtxSeed.czz;
308  if (!mat.InvertFast()) {
309  LOG(ERROR) << "Failed to invert matrix" << mat;
310  return false;
311  }
312  ROOT::Math::SVector<double, 3> rhs(vtxSeed.cx0, vtxSeed.cy0, vtxSeed.cz0);
313  auto sol = mat * rhs;
314  vtxSeed.setXYZ(sol(0), sol(1), sol(2));
315  if (vtxSeed.fillErrors) {
316  vtxSeed.setCov(mat(0, 0), mat(1, 0), mat(1, 1), mat(2, 0), mat(2, 1), mat(2, 2));
317  }
318  if (vtxSeed.tMeanAccErr > 0.) {
319  auto err2 = 1. / vtxSeed.tMeanAccErr;
320  vtxSeed.setTimeStamp({float(vtxSeed.tMeanAcc * err2), float(std::sqrt(err2))});
321  }
322 
323  vtxSeed.setChi2((vtxSeed.getNContributors() - vtxSeed.wghSum) / vtxSeed.scaleSig2ITuk2I); // calculate chi^2
324  auto newScale = vtxSeed.wghChi2 / vtxSeed.wghSum;
325  LOG(DEBUG) << "Solve: wghChi2=" << vtxSeed.wghChi2 << " wghSum=" << vtxSeed.wghSum << " -> scale= " << newScale << " old scale " << vtxSeed.scaleSigma2 << " prevScale: " << vtxSeed.scaleSigma2Prev;
326  //vtxSeed.print();
327  vtxSeed.setScale(newScale < mPVParams->minScale2 ? mPVParams->minScale2 : newScale, mTukey2I);
328  return true;
329 }

332 PVertexer::FitStatus PVertexer::evalIterations(VertexSeed& vtxSeed, PVertex& vtx) const
333 {
334  // decide if new iteration should be done, prepare next one if needed
335  // if scaleSigma2 reached its lower limit stop
336  PVertexer::FitStatus result = PVertexer::FitStatus::IterateFurther;
337 
338  if (vtxSeed.nIterations > mPVParams->maxIterations) {
339  result = PVertexer::FitStatus::Failure;
340  } else if (vtxSeed.scaleSigma2Prev <= mPVParams->minScale2 + kAlmost0F) {
341  result = PVertexer::FitStatus::OK;
342  LOG(DEBUG) << "stop on simga :" << vtxSeed.scaleSigma2 << " prev: " << vtxSeed.scaleSigma2Prev;
343  }
344  if (fair::Logger::Logging(fair::Severity::debug)) {
345  auto dchi = (vtx.getChi2() - vtxSeed.getChi2()) / vtxSeed.getChi2();
346  auto dx = vtxSeed.getX() - vtx.getX(), dy = vtxSeed.getY() - vtx.getY(), dz = vtxSeed.getZ() - vtx.getZ();
347  auto dst = std::sqrt(dx * dx + dy * dy + dz * dz);
348 
349  LOG(DEBUG) << "dChi:" << vtx.getChi2() << "->" << vtxSeed.getChi2() << " :-> " << dchi;
350  LOG(DEBUG) << "dx: " << dx << " dy: " << dy << " dz: " << dz << " -> " << dst;
351  }
352 
353  vtx = reinterpret_cast<const PVertex&>(vtxSeed);
354 
355  if (result == PVertexer::FitStatus::OK) {
356  auto chi2Mean = vtxSeed.getChi2() / vtxSeed.getNContributors();
357  if (chi2Mean > mPVParams->maxChi2Mean) {
358  result = PVertexer::FitStatus::Failure;
359  LOG(DEBUG) << "Rejecting at iteration " << vtxSeed.nIterations << " and ScalePrev " << vtxSeed.scaleSigma2Prev << " with meanChi2 = " << chi2Mean;
360  } else {
361  return result;
362  }
363  }
364 
365  if (vtxSeed.scaleSigma2 > vtxSeed.scaleSigma2Prev) {
366  if (++vtxSeed.nScaleIncrease > mPVParams->maxNScaleIncreased) {
367  result = PVertexer::FitStatus::Failure;
368  LOG(DEBUG) << "Rejecting at iteration " << vtxSeed.nIterations << " with NScaleIncreased " << vtxSeed.nScaleIncrease;
369  }
370  } else if (vtxSeed.scaleSigma2 > mPVParams->slowConvergenceFactor * vtxSeed.scaleSigma2Prev) {
371  if (++vtxSeed.nScaleSlowConvergence > mPVParams->maxNScaleSlowConvergence) {
372  if (vtxSeed.scaleSigma2 < mPVParams->acceptableScale2) {
373  vtxSeed.setScale(mPVParams->minScale2, mTukey2I);
374  LOG(DEBUG) << "Forcing scale2 to " << mPVParams->minScale2;
375  result = PVertexer::FitStatus::IterateFurther;
376  } else {
377  result = PVertexer::FitStatus::Failure;
378  LOG(DEBUG) << "Rejecting at iteration " << vtxSeed.nIterations << " with NScaleSlowConvergence " << vtxSeed.nScaleSlowConvergence;
379  }
380  }
381  } else {
382  vtxSeed.nScaleSlowConvergence = 0;
383  }
384 
385  return result;
386 }

388 //___________________________________________________________________
389 void PVertexer::initMeanVertexConstraint()
390 {
391  // set mean vertex constraint and its errors
392  double det = mMeanVertex.getSigmaY2() * mMeanVertex.getSigmaZ2() - mMeanVertex.getSigmaYZ() * mMeanVertex.getSigmaYZ();
393  if (det <= kAlmost0D || mMeanVertex.getSigmaY2() < kAlmost0D || mMeanVertex.getSigmaZ2() < kAlmost0D) {
394  throw std::runtime_error(fmt::format("Singular matrix for vertex constraint: syy={:+.4e} syz={:+.4e} szz={:+.4e}",
395  mMeanVertex.getSigmaY2(), mMeanVertex.getSigmaYZ(), mMeanVertex.getSigmaZ2()));
396  }
397  mXYConstraintInvErr[0] = mMeanVertex.getSigmaZ2() / det;
398  mXYConstraintInvErr[2] = mMeanVertex.getSigmaY2() / det;
399  mXYConstraintInvErr[1] = -mMeanVertex.getSigmaYZ() / det;
400 }

409 //___________________________________________________________________
410 TimeEst PVertexer::timeEstimate(const VertexingInput& input) const
411 {
412  o2::utils::StatAccumulator test;
413  for (int i : input.idRange) {
414  if (mTracksPool[i].canUse()) {
415  const auto& timeT = mTracksPool[i].timeEst;
416  auto trErr2I = 1. / (timeT.getTimeStampError() * timeT.getTimeStampError());
417  test.add(timeT.getTimeStamp(), trErr2I);
418  }
419  }
420  if (test.wsum > 0) {
421  float t = 0., te2 = 0.;
422  test.getMeanRMS2(t, te2);
423  return {t, te2};
424  } else {
425  return {0., 0.};
426  }
427 }

456 }

458 //___________________________________________________________________
459 void PVertexer::createTracksPool(gsl::span<const o2d::TrackTPCITS> tracksITSTPC)
460 {
461  // create pull of all candidate tracks in a global array ordered in time
462  mTracksPool.clear();
463  mSortedTrackID.clear();
464 
465  auto ntGlo = tracksITSTPC.size();
466  mTracksPool.reserve(ntGlo);
467  // check all containers
468  float vtxErr2 = 0.5 * (mMeanVertex.getSigmaX2() + mMeanVertex.getSigmaY2());
469  float dcaToler = 1.0 + 3. * std::sqrt(vtxErr2); // RS FIXME tolerance should be a parameter
470  float pullIniCut = 9.; // RS FIXME pullIniCut should be a parameter
471  o2d::DCA dca;
472 
473  for (uint32_t i = 0; i < ntGlo; i++) {
474  o2::track::TrackParCov trc = tracksITSTPC[i];
475  if (!trc.propagateToDCA(mMeanVertex, mBz, &dca, dcaToler) ||
476  dca.getY() * dca.getY() / (dca.getSigmaY2() + vtxErr2) > pullIniCut) {
477  continue;
478  }
479  auto& tvf = mTracksPool.emplace_back(trc, tracksITSTPC[i].getTimeMUS(), i, 0);
480  mStatZErr.add(std::sqrt(trc.getSigmaZ2()));
481  mStatTErr.add(tvf.timeEst.getTimeStampError());
482  }
483  // TODO: try to narrow timestamps using tof times
484  float zerrMean, zerrRMS;
485  mStatZErr.getMeanRMS2(zerrMean, zerrRMS);
486 
487  float terrMean, terrRMS;
488  mStatTErr.getMeanRMS2(terrMean, terrRMS);
489 
490  if (mTracksPool.empty()) {
491  return;

543 void PVertexer::clusterizeTime(float binSize, float maxTDist)
544 {
545  // moving average density
546  std::vector<int> tclistIdx, tcselSort;
547  std::vector<TimeCluster> tclist;
548  mTimesClusters.clear();
549 
550  int ntr = mTracksPool.size(), first = 0, last = 0, count = 1;
551  float tfirst = mTracksPool[mSortedTrackID[first]].timeEst.getTimeStamp(), tlast = tfirst, tMean = tfirst;
552  for (int i = 1; i < ntr; i++) {
553  if (mTracksPool[mSortedTrackID[i]].timeEst.getTimeStamp() - tfirst < binSize) {
554  last = i;
555  tlast = mTracksPool[mSortedTrackID[last]].timeEst.getTimeStamp();
556  tMean += tlast;
557  count++;
558  } else {
559  tclist.emplace_back(TimeCluster{{tMean / count, 0.}, first, last, count});
560  first = last = i;
561  tMean = tfirst = mTracksPool[mSortedTrackID[last]].timeEst.getTimeStamp();
562  count = 1;
563  }
564  }
565  tMean += mTracksPool[mSortedTrackID[last]].timeEst.getTimeStamp();
566  tclist.emplace_back(TimeCluster{{tMean / count, 0.}, first, last, count});
567 
568  int ncl = tclist.size();
569  tclistIdx.resize(ncl);
570  tcselSort.reserve(ncl);
571  std::iota(tclistIdx.begin(), tclistIdx.end(), 0);
572  std::sort(tclistIdx.begin(), tclistIdx.end(), [&tclist](int i, int j) { return tclist[i].count > tclist[j].count; });
573 
574  for (auto it : tclistIdx) {
575  auto& best = tclist[it];
576  if (!best.count) {
577  continue; // already merged
578  }
579  // check if it can absorm neighbours
580  int iBef = it, iAft = it;
581  bool merged;
582  do {
583  merged = false;
584  if (--iBef >= 0) {
585  if (tclist[iBef].count && best.isCompatible(tclist[iBef].timeEst, 0, maxTDist)) {
586  best.merge(tclist[iBef]);
587  merged = true;
588  } else {
589  iBef = -1; // don't check those before
590  }
591  }
592  if (++iAft < ncl) {
593  if (tclist[iAft].count && best.isCompatible(tclist[iAft].timeEst, 0, maxTDist)) {
594  best.merge(tclist[iAft]);
595  merged = true;
596  } else {
597  iAft = ncl; // don't check those after
598  }
599  }
600  } while (merged);
601  if (best.count > 1) { // RS FIXME
602  tcselSort.push_back(it);
603  }
604  }
605  // final sort in decreasing multiplicity
606  std::sort(tcselSort.begin(), tcselSort.end(), [&tclist](int i, int j) { return tclist[i].count > tclist[j].count; });
607  mTimesClusters.reserve(tcselSort.size());
608  for (auto i : tcselSort) {
609  mTimesClusters.push_back(tclist[i]);
610  // mTimesClusters.back().print();
611  }
612 }

615 void PVertexer::createMCLabels(gsl::span<const o2::MCCompLabel> lblITS, gsl::span<const o2::MCCompLabel> lblTPC,
616  const std::vector<PVertex> vertices, const std::vector<o2::dataformats::VtxTrackIndex> vertexTrackIDs, const std::vector<V2TRef> v2tRefs,
617  std::vector<o2::MCEventLabel>& lblVtx)
618 {
619  lblVtx.clear();
620  int nv = vertices.size();
621  if (lblITS.size() != lblITS.size() || !lblITS.size()) {
622  LOG(ERROR) << "labels are not provided or incorrect";
623  return;
624  }
625  std::unordered_map<o2::MCEventLabel, int> labelOccurenceCorr, labelOccurenceITS;
626 
627  auto bestLbl = [](std::unordered_map<o2::MCEventLabel, int> mp, int norm) -> o2::MCEventLabel {
628  o2::MCEventLabel best;
629  int bestCount = 0;
630  for (auto [lbl, cnt] : mp) {
631  if (cnt > bestCount) {
632  bestCount = cnt;
633  best = lbl;
634  }
635  }
636  if (bestCount && norm) {
637  best.setCorrWeight(float(bestCount) / norm);
638  }
639  return best;
640  };
641 
642  for (const auto& v2t : v2tRefs) {
643  int tref = v2t.getFirstEntry(), last = tref + v2t.getEntries();
644  labelOccurenceCorr.clear();
645  labelOccurenceITS.clear();
646  o2::MCEventLabel winner; // unset at the moment
647  for (; tref < last; tref++) {
648  int tid = vertexTrackIDs[tref].getIndex();
649  const auto& lITS = lblITS[tid];
650  const auto& lTPC = lblTPC[tid];
651  if (!lITS.isSet() || !lTPC.isSet()) {
652  break;
653  }
654  if (lITS.getTrackID() == lITS.getTrackID() && lITS.getEventID() == lTPC.getEventID() && lITS.getSourceID() == lTPC.getSourceID()) {
655  labelOccurenceCorr[{lITS.getEventID(), lITS.getSourceID(), 0.}]++;
656  } else {
657  labelOccurenceITS[{lITS.getEventID(), lITS.getSourceID(), 0.}]++;
658  }
659  }
660  if (labelOccurenceCorr.size()) {
661  winner = bestLbl(labelOccurenceCorr, v2t.getEntries());
662  } else if (labelOccurenceITS.size()) {
663  winner = bestLbl(labelOccurenceITS, 0); // in absence of correct matches, set the ITS only label but set its weight to 0
664  }
665  lblVtx.push_back(winner);
666  }
667 }

670 std::pair<int, int> PVertexer::getBestFT0Trigger(const PVertex& vtx, gsl::span<const o2::ft0::RecPoints> ft0Data, int& currEntry) const
671 {
672  // select best matching FT0 recpoint
673  int best = -1, n = ft0Data.size();
674  while (currEntry < n && ft0Data[currEntry].getInteractionRecord() < vtx.getIRMin()) {
675  currEntry++; // skip all times which have no chance to be matched
676  }
677  int i = currEntry, nCompatible = 0;
678  float bestDf = 1e12;
679  auto tVtxNS = (vtx.getTimeStamp().getTimeStamp() + mPVParams->timeBiasMS) * 1e3; // time in ns
680  while (i < n) {
681  if (ft0Data[i].getInteractionRecord() > vtx.getIRMax()) {
682  break;
683  }
684  if (mFT0Params->isSelected(ft0Data[i])) {
685  nCompatible++;
686  auto dfa = std::abs(mFT0Params->getInteractionTimeNS(ft0Data[i], mStartIR) - tVtxNS);
687  if (dfa <= bestDf) {
688  bestDf = dfa;
689  best = i;
690  }
691  }
692  i++;
693  }
694  return {best, nCompatible};
695 }

722 //___________________________________________________________________
723 bool PVertexer::setCompatibleIR(PVertex& vtx)
724 {
725  // assign compatible IRs accounting for the bunch filling scheme
726  const auto& vtxT = vtx.getTimeStamp();
727  o2::InteractionRecord irMin(mStartIR), irMax(mStartIR);
728  auto rangeT = mPVParams->nSigmaTimeCut * std::max(mPVParams->minTError, std::min(mPVParams->maxTError, vtxT.getTimeStampError()));
729  float t = vtxT.getTimeStamp() + mPVParams->timeBiasMS;
730  if (t > rangeT) {
731  irMin += o2::InteractionRecord(1.e3 * (t - rangeT));
732  }
733  irMax += o2::InteractionRecord(1.e3 * (t + rangeT));
734  irMax++; // to account for rounding
735  // restrict using bunch filling
736  int bc = mClosestBunchAbove[irMin.bc];
737  if (bc < irMin.bc) {
738  irMin.orbit++;
739  }
740  irMin.bc = bc;
741  bc = mClosestBunchBelow[irMax.bc];
742  if (bc > irMax.bc) {
743  if (irMax.orbit == 0) {
744  return false;
745  }
746  irMax.orbit--;
747  }
748  irMax.bc = bc;
749  vtx.setIRMin(irMin);
750  vtx.setIRMax(irMax);
751  return irMax >= irMin;
752 }
