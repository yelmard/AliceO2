42 #define makelogaxis(h) \
43  { \
44  const Int_t nbins = h->GetNbinsX(); \
45  double binp[nbins + 1]; \
46  double max = h->GetXaxis()->GetBinUpEdge(nbins); \
47  double min = h->GetXaxis()->GetBinLowEdge(1); \
48  double lmin = TMath::Log10(min); \
49  double ldelta = (TMath::Log10(max) - lmin) / ((double)nbins); \
50  for (int i = 0; i < nbins; i++) { \
51  binp[i] = TMath::Exp(TMath::Log(10) * (lmin + i * ldelta)); \
52  } \
53  binp[nbins] = max + 1; \
54  h->GetXaxis()->Set(nbins, binp); \
55  }
