32 void DetectorResponse::LoadParamFromFile(const TString fname, const TString pname, const Param_t ptype)
33 {
34  TFile f(fname, "READ");
35  if (!f.Get(pname)) {
36  LOG(fatal) << "Did not find parametrization " << pname << " in file " << fname;
37  }
38  LOG(info) << "Loading parametrization " << pname << " from TFile " << fname;
39  f.GetObject(pname, mParam[ptype]);
40  f.Close();
41  mParam[ptype]->Print();
42  mParam[ptype]->PrintParametrization();
43 }
