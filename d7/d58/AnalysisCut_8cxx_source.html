59 bool AnalysisCut::IsSelected(float* values)
60 {
61  //
62  // apply the configured cuts
63  //
64  // iterate over cuts
65  for (std::vector<CutContainer>::iterator it = fCuts.begin(); it != fCuts.end(); ++it) {
66  // check whether a dependent variables were enabled and if they are in the requested range
67  if ((*it).fDepVar != -1) {
68  bool inRange = (values[(*it).fDepVar] > (*it).fDepLow && values[(*it).fDepVar] <= (*it).fDepHigh);
69  if (!inRange && !((*it).fDepExclude))
70  continue;
71  if (inRange && (*it).fDepExclude)
72  continue;
73  }
74  if ((*it).fDepVar2 != -1) {
75  bool inRange = (values[(*it).fDepVar2] > (*it).fDep2Low && values[(*it).fDepVar2] <= (*it).fDep2High);
76  if (!inRange && !((*it).fDep2Exclude))
77  continue;
78  if (inRange && (*it).fDep2Exclude)
79  continue;
80  }
81  // obtain the low and high cut values (either directly as a value or from a function)
82  float cutLow, cutHigh;
83  if ((*it).fFuncLow)
84  cutLow = ((*it).fFuncLow)->Eval(values[(*it).fDepVar]);
85  else
86  cutLow = ((*it).fLow);
87  if ((*it).fFuncHigh)
88  cutHigh = ((*it).fFuncHigh)->Eval(values[(*it).fDepVar]);
89  else
90  cutHigh = ((*it).fHigh);
91  // apply the cut and return the decision
92  bool inRange = (values[(*it).fVar] >= cutLow && values[(*it).fVar] <= cutHigh);
93  if (!inRange && !((*it).fExclude))
94  return false;
95  if (inRange && ((*it).fExclude))
96  return false;
97  }
98 
99  return true;
100 }
