38 // This is a very simple example showing how to iterate over tracks
39 // and create a new collection for them.
40 // FIXME: this should really inherit from AnalysisTask but
41 // we need GCC 7.4+ for that
42 struct ATask {
43  Produces<aod::EtaPhi> etaphi;
44 
45  void process(aod::Tracks const& tracks)
46  {
47  for (auto& track : tracks) {
48  float phi = asin(track.snp()) + track.alpha() + static_cast<float>(M_PI);
49  float eta = log(tan(0.25f * static_cast<float>(M_PI) - 0.5f * atan(track.tgl())));
50  etaphi(eta, phi);
51  }
52  }
53 };

64 struct BTask {

67  LOGF(INFO, "ID: %d", collision.globalIndex());
