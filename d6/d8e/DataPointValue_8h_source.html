384  inline void set_payload(
385  const uint64_t* const data,
386  const DeliveryType type)
387  {
388  switch (type) {
389  case RAW_INT:
390  case DPVAL_INT:
391  case RAW_UINT:
392  case DPVAL_UINT:
393  this->payload_pt1 = data[0] & 0x00000000FFFFFFFF;
394  break;
395  case RAW_BOOL:
396  case DPVAL_BOOL:
397  this->payload_pt1 = (data[0] ? 1 : 0);
398  break;
399  case RAW_CHAR:
400  case DPVAL_CHAR:
401  this->payload_pt1 = data[0] & 0x00000000000000FF;
402  break;
403  case RAW_DOUBLE:
404  case DPVAL_DOUBLE:
405  case RAW_TIME:
406  case DPVAL_TIME:
407  this->payload_pt1 = data[0];
408  break;
409  case RAW_STRING:
410  case DPVAL_STRING:
411  std::strncpy((char*)&payload_pt1, (char*)data, 56);
412  break;
413  case RAW_BINARY:
414  case DPVAL_BINARY:
415  memcpy((void*)&payload_pt1, (void*)data, 56);
416  case VOID:
417  break;
418  default:
419  throw std::domain_error("Invalid DeliveryType.");
420  }
