132  void process(soa::Filtered<soa::Join<aod::FullTracks, aod::TracksExtended, aod::TrackSelection>>::iterator const& track)
133  {
134  // fill kinematic variables
135  kine->Fill(pt, track.pt());
136  kine->Fill(eta, track.eta());
137  kine->Fill(phi, track.phi());
138 
139  // fill track parameters
140  trackpar->Fill(alpha, track.alpha());
141  trackpar->Fill(x, track.x());
142  trackpar->Fill(y, track.y());
143  trackpar->Fill(z, track.z());
144  trackpar->Fill(signed1Pt, track.signed1Pt());
145  trackpar->Fill(snp, track.snp());
146  trackpar->Fill(tgl, track.tgl());
147  for (unsigned int i = 0; i < 64; i++) {
148  if (track.flags() & (1 << i))
149  trackpar->Fill(flags, i);
150  }
151  trackpar->Fill(dcaXY, track.dcaXY());
152  trackpar->Fill(dcaZ, track.dcaZ());
153 
154  // fill ITS variables
155  its->Fill(itsNCls, track.itsNCls());
156  its->Fill(itsChi2NCl, track.itsChi2NCl());
157  for (unsigned int i = 0; i < 7; i++) {
158  if (track.itsClusterMap() & (1 << i))
159  its->Fill(itsHits, i);
160  }
161 
162  // fill TPC variables
163  tpc->Fill(tpcNClsFindable, track.tpcNClsFindable());
164  tpc->Fill(tpcNClsFound, track.tpcNClsFound());
165  tpc->Fill(tpcNClsShared, track.tpcNClsShared());
166  tpc->Fill(tpcNClsCrossedRows, track.tpcNClsCrossedRows());
167  tpc->Fill(tpcCrossedRowsOverFindableCls, track.tpcCrossedRowsOverFindableCls());
168  tpc->Fill(tpcFractionSharedCls, track.tpcFractionSharedCls());
169  tpc->Fill(tpcChi2NCl, track.tpcChi2NCl());
170 
171  // fill TRD variables
172 
173  // fill TOF variables
174 
175  // fill EMCAL variables
176  }
177 };
